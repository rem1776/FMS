114  !> @brief This statistics is for bilinear interpolation and spherical regrid.
115  subroutine stats ( dat, low, high, avg, miss, missing_value, mask )
116  real, intent(in) :: dat(:,:)
117  real, intent(out) :: low, high, avg
118  integer, intent(out) :: miss
119  real, intent(in), optional :: missing_value
120  real, intent(in), optional :: mask(:,:)
121 
122  real :: dsum, npts, buffer_real(3)
123  integer :: pe, root_pe, npes, p, buffer_int(2)
124 
125  pe = mpp_pe()
126  root_pe = mpp_root_pe()
127  npes = mpp_npes()
128 
129  dsum = 0.0
130  miss = 0
131 
132  if (present(missing_value)) then
133  miss = count(dat(:,:) == missing_value)
134  low = minval(dat(:,:), dat(:,:) /= missing_value)
135  high = maxval(dat(:,:), dat(:,:) /= missing_value)
136  dsum = sum(dat(:,:), dat(:,:) /= missing_value)
137  else if(present(mask)) then
138  miss = count(mask(:,:) <= 0.5)
139  low = minval(dat(:,:),mask=mask(:,:) > 0.5)
140  high = maxval(dat(:,:),mask=mask(:,:) > 0.5)
141  dsum = sum(dat(:,:), mask=mask(:,:) > 0.5)
142  else
143  miss = 0
144  low = minval(dat(:,:))
145  high = maxval(dat(:,:))
146  dsum = sum(dat(:,:))
147  endif
148  avg = 0.0
149 
150  npts = size(dat(:,:)) - miss
151  if(pe == root_pe) then
152  do p = 1, npes - 1 ! root_pe receive data from other pe
153  ! Force use of "scalar", integer pointer mpp interface
154  call mpp_recv(buffer_real(1),glen=3, from_pe=p+root_pe, tag=comm_tag_1)
155  dsum = dsum + buffer_real(1)
156  low = min(low, buffer_real(2))
157  high = max(high, buffer_real(3))
158  call mpp_recv(buffer_int(1), glen=2, from_pe=p+root_pe, tag=comm_tag_2)
159  miss = miss + buffer_int(1)
160  npts = npts + buffer_int(2)
161  enddo
162  if(npts == 0.) then
163  print*, 'Warning: no points is valid'
164  else
165  avg = dsum/real(npts)
166  endif
167  else ! other pe send data to the root_pe.
168  buffer_real(1) = dsum
169  buffer_real(2) = low
170  buffer_real(3) = high
171  ! Force use of "scalar", integer pointer mpp interface
172  call mpp_send(buffer_real(1),plen=3,to_pe=root_pe, tag=comm_tag_1)
173  buffer_int(1) = miss
174  buffer_int(2) = npts
175  call mpp_send(buffer_int(1), plen=2, to_pe=root_pe, tag=comm_tag_2)
176  endif
177 
178  call mpp_sync_self()
179 
180  return
181 
182  end subroutine stats

60 !> Holds data pointers and metadata for horizontal interpolations, passed between the horiz_interp modules
61 !> @ingroup horiz_interp_type_mod
62  type :: horiz_interp_type
63  integer, dimension(:,:), pointer :: ilon =>null() !< indices for conservative scheme
64  integer, dimension(:,:), pointer :: jlat =>null() !< indices for conservative scheme
65  integer, dimension(:,:,:), pointer :: i_lon =>null() !< indices for bilinear interpolation
66  !! and spherical regrid
67  integer, dimension(:,:,:), pointer :: j_lat =>null() !< indices for bilinear interpolation
68  !! and spherical regrid
69  logical, dimension(:,:), pointer :: found_neighbors =>null() !< indicate whether destination grid
70  !! has some source grid around it.
71  integer, dimension(:,:), pointer :: num_found => null()
72  integer :: nlon_src !< size of source grid
73  integer :: nlat_src !< size of source grid
74  integer :: nlon_dst !< size of destination grid
75  integer :: nlat_dst !< size of destination grid
76  integer :: interp_method !< interpolation method.
77  !! =1, conservative scheme
78  !! =2, bilinear interpolation
79  !! =3, spherical regrid
80  !! =4, bicubic regrid
81  logical :: i_am_initialized=.false.
82  integer :: version !< indicate conservative
83  !! interpolation version with value 1 or 2
84  !--- The following are for conservative interpolation scheme version 2 ( through xgrid)
85  integer :: nxgrid !< number of exchange grid
86  !! between src and dst grid.
87  integer, dimension(:), pointer :: i_src=>null() !< indices in source grid.
88  integer, dimension(:), pointer :: j_src=>null() !< indices in source grid.
89  integer, dimension(:), pointer :: i_dst=>null() !< indices in destination grid.
90  integer, dimension(:), pointer :: j_dst=>null() !< indices in destination grid.
91  type(horiz_interp_reals8), allocatable :: kind8_reals !< derived type holding kind 8 real data pointers if used
92  type(horiz_interp_reals4), allocatable :: kind4_reals !< derived type holding kind 4 real data pointers if used
93 
94  end type

96 !> real(8) pointers for use in horiz_interp_type
97 type horiz_interp_reals8
98  real(kind=r8_kind), dimension(:,:), pointer :: faci =>null() !< weights for conservative scheme
99  real(kind=r8_kind), dimension(:,:), pointer :: facj =>null() !< weights for conservative scheme
100  real(kind=r8_kind), dimension(:,:), pointer :: area_src =>null() !< area of the source grid
101  real(kind=r8_kind), dimension(:,:), pointer :: area_dst =>null() !< area of the destination grid
102  real(kind=r8_kind), dimension(:,:,:), pointer :: wti =>null() !< weights for bilinear interpolation
103  !! wti ist used for derivative "weights" in bicubic
104  real(kind=r8_kind), dimension(:,:,:), pointer :: wtj =>null() !< weights for bilinear interpolation
105  !! wti ist used for derivative "weights" in bicubic
106  real(kind=r8_kind), dimension(:,:,:), pointer :: src_dist =>null() !< distance between destination grid and
107  !! neighbor source grid.
108  real(kind=r8_kind), dimension(:,:), pointer :: rat_x =>null() !< the ratio of coordinates of the dest grid
109  !! (x_dest -x_src_r)/(x_src_l -x_src_r)
110  !! and (y_dest -y_src_r)/(y_src_l -y_src_r)
111  real(kind=r8_kind), dimension(:,:), pointer :: rat_y =>null() !< the ratio of coordinates of the dest grid
112  !! (x_dest -x_src_r)/(x_src_l -x_src_r)
113  !! and (y_dest -y_src_r)/(y_src_l -y_src_r)
114  real(kind=r8_kind), dimension(:), pointer :: lon_in =>null() !< the coordinates of the source grid
115  real(kind=r8_kind), dimension(:), pointer :: lat_in =>null() !< the coordinates of the source grid
116  real(kind=r8_kind), dimension(:), pointer :: area_frac_dst=>null() !< area fraction in destination grid.
117  real(kind=r8_kind), dimension(:,:), pointer :: mask_in=>null()
118  real(kind=r8_kind) :: max_src_dist
119 
120 end type horiz_interp_reals8

50 subroutine mpp_scatter_pelist_3d_(is, ie, js, je, nk, pelist, array_seg, data, is_root_pe, &
51  ishift, jshift)
52  integer, intent(in) :: is, ie, js, je, nk
53  integer, dimension(:), intent(in) :: pelist
54  mpp_type_, dimension(is:ie,js:je,1:nk), intent(inout) :: array_seg
55  mpp_type_, dimension(:,:,:), intent(in) :: data
56  logical, intent(in) :: is_root_pe
57  integer, optional, intent(in) :: ishift, jshift
58 
59  integer :: i, msgsize, root_pe, root_pe_test
60  integer :: i1, i2, j1, j2, ioff, joff
61  integer :: my_ind(4), gind(4,size(pelist))
62  type array3d
63  mpp_type_, dimension(:,:,:), allocatable :: data
64  endtype array3d
65  type(array3d), dimension(size(pelist)) :: temp
66 
67  if (.not.any(mpp_pe().eq.pelist(:))) return
68 
69  if (is_root_pe) then
70  root_pe = mpp_pe()
71  root_pe_test = 999
72  if (.not.any(pelist(:).eq.root_pe)) call mpp_error(fatal, &
73  "fms_io(mpp_scatter_pelist): root_pe not a member of pelist")
74  else
75  root_pe = 0
76  root_pe_test = -999
77  endif
78 ! need this check in case MPI-rank 0 is a member of the pelist
79  call mpp_max(root_pe_test, pelist)
80  if (root_pe_test.lt.0) call mpp_error(fatal, &
81  "fms_io(mpp_scatter_pelist): root_pe not specified or not a member of the pelist")
82 ! need to make sure only one root_pe has been specified
83  call mpp_sum(root_pe, pelist)
84  if ((is_root_pe) .and. (mpp_pe().ne.root_pe)) call mpp_error(fatal, &
85  "fms_io(mpp_scatter_pelist): too many root_pes specified")
86 
87 
88  ioff=0
89  joff=0
90  if (present(ishift)) ioff=ishift
91  if (present(jshift)) joff=jshift
92 
93  my_ind(1) = is
94  my_ind(2) = ie
95  my_ind(3) = js
96  my_ind(4) = je
97 
98 ! scatter indices into global index on root_pe
99  if (is_root_pe) then
100  do i = 1, size(pelist)
101 ! root_pe data copy - no send to self
102  if (pelist(i).eq.root_pe) then
103  gind(:,i) = my_ind(:)
104  else
105  call mpp_recv(gind(:,i:i), 4, pelist(i), .false., comm_tag_1)
106  endif
107  enddo
108  call mpp_sync_self(check=event_recv)
109  gind(1,:)=gind(1,:)+ioff
110  gind(2,:)=gind(2,:)+ioff
111  gind(3,:)=gind(3,:)+joff
112  gind(4,:)=gind(4,:)+joff
113 ! check indices to make sure they are within the range of "data"
114  if ((minval(gind).lt.1) .OR. (maxval(gind(1:2,:)).gt.size(data,1)) .OR. (maxval(gind(3:4,:)).gt.size(data,2))) &
115  call mpp_error(fatal,"fms_io(mpp_scatter_pelist): specified indices (with shift) are outside of the &
116  &range of the receiving array")
117  else
118 ! non root_pe's send indices to root_pe
119  call mpp_send(my_ind(:), 4, root_pe, comm_tag_1)
120  call mpp_sync_self(check=event_send)
121  endif
122 
123 ! scatter segments into data based on indices
124  if (is_root_pe) then
125  do i = 1, size(pelist)
126  if (pelist(i).ne.root_pe) then ! no send to self
127  i1 = gind(1,i)
128  i2 = gind(2,i)
129  j1 = gind(3,i)
130  j2 = gind(4,i)
131  msgsize = (i2-i1+1)*(j2-j1+1)*nk
132 ! allocate and copy data into a contiguous memory space
133  allocate(temp(i)%data(i1:i2,j1:j2,1:nk))
134  temp(i)%data(i1:i2,j1:j2,1:nk)=data(i1:i2,j1:j2,1:nk)
135  call mpp_send(temp(i)%data, msgsize, pelist(i), comm_tag_2)
136  else
137 ! data copy - no send to self
138  array_seg(is:ie,js:je,1:nk) = data(is+ioff:ie+ioff,js+joff:je+joff,1:nk)
139  endif
140  enddo
141  call mpp_sync_self(check=event_send)
142 ! deallocate the temporary array used for the send
143  do i = 1, size(pelist)
144  if (allocated(temp(i)%data)) deallocate(temp(i)%data)
145  enddo
146  else
147 ! non root_pe's recv data from root_pe
148  msgsize = (my_ind(2)-my_ind(1)+1) * (my_ind(4)-my_ind(3)+1) * nk
149  call mpp_recv(array_seg, msgsize, root_pe, .false., comm_tag_2)
150  call mpp_sync_self(check=event_recv)
151  endif
152 
153  call mpp_sync_self()
154 
155  return
156 
157 end subroutine mpp_scatter_pelist_3d_

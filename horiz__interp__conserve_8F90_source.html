19 !> @defgroup horiz_interp_conserve_mod horiz_interp_conserve_mod
20 !> @ingroup horiz_interp
21 !> @brief Performs spatial interpolation between grids using conservative interpolation
22 !!
23 !> @author Bruce Wyman, Zhi Liang
24 !!
25 !> This module can conservatively interpolate data from any logically rectangular grid
26 !! to any rectangular grid. The interpolation scheme is area-averaging
27 !! conservative scheme. There is an optional mask field for missing input data in both
28 !! horiz_interp__conserveinit and horiz_interp_conserve. For efficiency purpose, mask should only be
29 !! kept in horiz_interp_init (will remove the mask in horiz_interp in the future).
30 !! There are 1-D and 2-D version of horiz_interp_conserve_init for 1-D and 2-D grid.
31 !! There is a optional argument mask in horiz_interp_conserve_init_2d and no mask should
32 !! to passed into horiz_interp_conserv. optional argument mask will not be passed into
33 !! horiz_interp_conserve_init_1d and optional argument mask may be passed into
34 !! horiz_interp_conserve (For the purpose of reproduce Memphis??? results).
35 !! An optional output mask field may be used in conjunction with the input mask to show
36 !! where output data exists.

136  !> Deallocates memory used by "HI_KIND_TYPE" variables.

138  subroutine horiz_interp_conserve_del ( Interp )
139 
140  type (horiz_interp_type), intent(inout) :: interp !< A derived-type variable returned by
141  !! previous call to horiz_interp_new. The input variable must have
142  !! allocated arrays. The returned variable will contain deallocated arrays.
143 
144  select case(interp%version)
145  case (1)
146  if( allocated( interp%kind8_reals)) then
147  if(associated(interp%kind8_reals%area_src)) deallocate(interp%kind8_reals%area_src)

149  if(associated(interp%kind8_reals%facj)) deallocate(interp%kind8_reals%facj)
150  if(associated(interp%jlat)) deallocate(interp%jlat)
151  if(associated(interp%kind8_reals%faci)) deallocate(interp%kind8_reals%faci)
152  if(associated(interp%ilon)) deallocate(interp%ilon)
153  else if( allocated( interp%kind4_reals)) then
154  if(associated(interp%kind4_reals%area_src)) deallocate(interp%kind4_reals%area_src)
155  if(associated(interp%kind4_reals%area_dst)) deallocate(interp%kind4_reals%area_dst)
156  if(associated(interp%kind4_reals%facj)) deallocate(interp%kind4_reals%facj)
157  if(associated(interp%jlat)) deallocate(interp%jlat)
158  if(associated(interp%kind4_reals%faci)) deallocate(interp%kind4_reals%faci)
159  if(associated(interp%ilon)) deallocate(interp%ilon)
160  endif
161  case (2)
162  if( allocated( interp%kind8_reals)) then
163  if(associated(interp%i_src)) deallocate(interp%i_src)
164  if(associated(interp%j_src)) deallocate(interp%j_src)
165  if(associated(interp%i_dst)) deallocate(interp%i_dst)
166  if(associated(interp%j_dst)) deallocate(interp%j_dst)
167  if(associated(interp%kind8_reals%area_frac_dst)) deallocate(interp%kind8_reals%area_frac_dst)
168  else if( allocated( interp%kind4_reals)) then
169  if(associated(interp%i_src)) deallocate(interp%i_src)
170  if(associated(interp%j_src)) deallocate(interp%j_src)
171  if(associated(interp%i_dst)) deallocate(interp%i_dst)
172  if(associated(interp%j_dst)) deallocate(interp%j_dst)
173  if(associated(interp%kind4_reals%area_frac_dst)) deallocate(interp%kind4_reals%area_frac_dst)
174  endif
175  end select

20 !ug support
21 
22 !>For an inputted file name, check if it or any of its variants exist.
23 !!For a file named "foo", variants checked (in order) include:
24 !!
25 !! foo
26 !! foo.nc
27 !! foo.<domain_tile_id_string>.nc
28 !! foo.nc.<IO_domain_tile_id_string>
29 !! foo.<domain_tile_id_string>.nc.<IO_domain_tile_id_string>
30 !! foo.<ensemble_id>
31 !! foo.<ensemble_id>.nc
32 !! foo.<ensemble_id>.<domain_tile_id_string>.nc
33 !! foo.<ensemble_id>.nc.<IO_domain_tile_id_string>
34 !! foo.<ensemble_id>.<domain_tile_id_string>.nc.<IO_domain_tile_id_string>
35 !!
36 !!If a match is found, the value true is returned for the "does_file_exist"
37 !!flag. In addition, the actual file name is returned and the "read_dist"
38 !!flag, which tells whether or not the filename contains the
39 !!IO_domain_tile_id_string appended.
40 !!
41 !!Should this be a subroutine instead of a funtion for clarity since it
42 !!returns more than one value?
43 function fms_io_unstructured_get_file_name(orig_file, &
44  actual_file, &
45  read_dist, &
46  domain) &
47  result(does_file_exist)
48 
49  !Inputs/Outputs
50  character(len=*),intent(in) :: orig_file !<The name of file we're looking for.
51  character(len=*),intent(out) :: actual_file !<Name of the file we found.
52  logical(INT_KIND),intent(out) :: read_dist !<Flag telling if the file is "distributed" (has IO domain tile id appended onto the end).
53  type(domainUG),intent(in) :: domain !<Unstructured mpp domain.
54  logical(INT_KIND) :: does_file_exist !<Flag telling if the inputted file exists or one its variants.
55 
56  !Local variables
57  logical(INT_KIND) :: fexist !<Flag that tells if a file exists.
58  type(domainUG),pointer :: io_domain !<Pointer to an unstructured I/O domain.
59  integer(INT_KIND) :: io_tile_id !<Tile id for the I/O domain.
60  character(len=256) :: fname !<A character buffer used to test different file names.
61  character(len=512) :: actual_file_tmp !<A character buffer used to test different file names.
62 
63  !Set the default return values for the function.
64  actual_file = ""
65  does_file_exist = .false.
66  read_dist = .false.
67 
68  !Check if the file name does not contain ".nc".
69  fexist = .false.
70  if (index(orig_file,".nc",back=.true.) .eq. 0) then
71  inquire(file=trim(orig_file),exist=fexist)
72  if (fexist) then
73  actual_file = orig_file
74  does_file_exist = .true.
75  return
76  endif
77  endif
78 
79  !If necessary, add the correct domain ".tilexxxx" string to the inputted
80  !file name. For a file named foo.nc, this would become foo.tilexxxx.nc.
81  !Check if the new file name exists.
82  call get_mosaic_tile_file_ug(orig_file, &
83  actual_file, &
84  domain)
85  inquire(file=trim(actual_file),exist=fexist)
86  if (fexist) then
87  does_file_exist = .true.
88  return
89  endif
90 
91  !Point to the I/O domain for the unstructured grid. This function call
92  !will throw a fatal error if the I/O domain does not exist.
93  io_domain => null()
94  io_domain => mpp_get_UG_io_domain(domain)
95 
96  !Get the tile id for the I/O domain.
97  io_tile_id = mpp_get_UG_domain_tile_id(io_domain)
98  io_domain => null()
99 
100  !Check if the file has the I/O domain's tile id appended to the end of its
101  !name. For a file named foo.nc, this would become foo.nc.yyyy, where
102  !"yyyy" would in reality be the I/O domain's tile id. If the file exists,
103  !then set the read_dist and does_file_exist flags to true and return.
104  write(fname,'(a,i4.4)') trim(actual_file)//'.',io_tile_id
105  inquire(file=trim(fname),exist=fexist)
106  if (.not. fexist) then
107  write(fname,'(a,i6.6)') trim(actual_file)//'.',io_tile_id
108  inquire(file=trim(fname),exist=fexist)
109  endif
110  if (fexist) then
111  read_dist = .true.
112  does_file_exist = .true.
113  return
114  endif
115 
116  !Check if the file is part of an ensemble.
117  !filename_appendix is a module variable.
118  if (len_trim(filename_appendix) .gt. 0) then
119  call get_instance_filename(orig_file, &
120  actual_file)
121  if (index(orig_file,'.nc',back=.true.) .eq. 0) then
122  inquire(file=trim(actual_file),exist=fexist)
123  if (fexist) then
124  does_file_exist = .true.
125  return
126  endif
127  endif
128 
129  !Make a local copy of "actual_file", and the use the local copy to
130  !add the domain ".tilexxxx" string to "actual_file".
131  actual_file_tmp = actual_file
132  call get_mosaic_tile_file_ug(actual_file_tmp, &
133  actual_file, &
134  domain)
135  inquire(file=trim(actual_file),exist=fexist)
136  if (fexist) then
137  does_file_exist = .true.
138  return
139  endif
140 
141  !Point to the I/O domain for the unstructured grid. This function call
142  !will throw a fatal error if the I/O domain does not exist.
143  io_domain => mpp_get_UG_io_domain(domain)
144 
145  !Get the tile id for the I/O domain.
146  io_tile_id = mpp_get_UG_domain_tile_id(io_domain)
147  io_domain => null()
148 
149  !Check if the file has the I/O domain's tile id appended to the end of
150  !its name. If it does then set the read_dist and does_file_exist flags
151  !to true and return.
152  write(fname,'(a,i4.4)') trim(actual_file)//'.',io_tile_id
153  inquire(file=trim(fname),exist=fexist)
154  if (.not. fexist) then
155  write(fname,'(a,i6.6)') trim(actual_file)//'.',io_tile_id
156  inquire(file=trim(fname),exist=fexist)
157  endif
158  if (fexist) then
159  read_dist = .true.
160  does_file_exist = .true.
161  return
162  endif
163  endif
164 
165  return
166 end function fms_io_unstructured_get_file_name

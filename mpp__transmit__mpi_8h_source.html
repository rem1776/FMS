25 !> A message-passing routine intended to be reminiscent equally of both MPI and SHMEM
26 !! put_data and get_data are contiguous MPP_TYPE_ arrays
27 !!at each call, your put_data array is put to to_pe's get_data
28 !! your get_data array is got from from_pe's put_data
29 !!i.e we assume that typically (e.g updating halo regions) each PE performs a put _and_ a get
30 !!special PE designations:
31 !! NULL_PE: to disable a put or a get (e.g at boundaries)
32 !! ANY_PE: if remote PE for the put or get is to be unspecific
33 !! ALL_PES: broadcast and collect operations (collect not yet implemented)
34 !!ideally we would not pass length, but this f77-style call performs better (arrays passed by address, not descriptor)
35 !!further, this permits <length> contiguous words from an array of any rank to be passed (avoiding f90 rank conformance check)
36 !!caller is responsible for completion checks (mpp_sync_self) before and after
37  subroutine mpp_transmit_( put_data, put_len, to_pe, get_data, get_len, from_pe, block, tag, recv_request, send_request )
38 
39  integer, intent(in) :: put_len, to_pe, get_len, from_pe
40  mpp_type_, intent(in) :: put_data(*)
41  mpp_type_, intent(out) :: get_data(*)
42  logical, intent(in), optional :: block
43  integer, intent(in), optional :: tag
44  integer, intent(out), optional :: recv_request, send_request
45  logical :: block_comm
46  integer :: i
47  mpp_type_, allocatable, save :: local_data(:) !local copy used by non-parallel code (no SHMEM or MPI)
48  integer :: comm_tag
49  integer :: rsize
50 
51  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_TRANSMIT: You must first call mpp_init.' )
52  if( to_pe.EQ.null_pe .AND. from_pe.EQ.null_pe )return
53 
54  block_comm = .true.
55  if(PRESENT(block)) block_comm = block
56 
57  if( debug )then
58  call system_clock(tick)
59  write( stdout_unit,'(a,i18,a,i6,a,2i6,2i8)' )&
60  'T=',tick, ' PE=',pe, ' MPP_TRANSMIT begin: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
61  end if
62 
63  comm_tag = default_tag
64  if(present(tag)) comm_tag = tag
65 
66 !do put first and then get
67  if( to_pe.GE.0 .AND. to_pe.LT.npes )then
68 !use non-blocking sends
69  if( debug .and. (current_clock.NE.0) )call system_clock(start_tick)
70 !z1l: truly non-blocking send.
71 ! if( request(to_pe).NE.MPI_REQUEST_NULL )then !only one message from pe->to_pe in queue
72 ! if( debug )write( stderr(),* )'PE waiting for sending', pe, to_pe
73 ! call MPI_WAIT( request(to_pe), stat, error )
74 ! end if
75  if(present(send_request)) then
76  call mpi_isend( put_data, put_len, mpi_type_, to_pe, comm_tag, mpp_comm_private, send_request, error)
77  else
78  cur_send_request = cur_send_request + 1
79  if( cur_send_request > max_request ) call mpp_error(fatal, &
80  "MPP_TRANSMIT: cur_send_request is greater than max_request, increase mpp_nml request_multiply")
81  call mpi_isend( put_data, put_len, mpi_type_, to_pe, comm_tag, mpp_comm_private, &
82  request_send(cur_send_request), error)
83  endif
84  if( debug .and. (current_clock.NE.0) )call increment_current_clock( event_send, put_len*mpp_type_bytelen_ )
85  else if( to_pe.EQ.all_pes )then !this is a broadcast from from_pe
86  if( from_pe.LT.0 .OR. from_pe.GE.npes )call mpp_error( fatal, 'MPP_TRANSMIT: broadcasting from invalid PE.' )
87  if( put_len.GT.get_len )call mpp_error( fatal, 'MPP_TRANSMIT: size mismatch between put_data and get_data.' )
88  if( pe.EQ.from_pe )then
89  if( loc(get_data).NE.loc(put_data) )then
90 !dir$ IVDEP
91  do i = 1,get_len
92  get_data(i) = put_data(i)
93  end do
94  end if
95  end if
96  call mpp_broadcast( get_data, get_len, from_pe )
97  return
98  else if( to_pe.EQ.any_pe )then !we don't have a destination to do puts to, so only do gets
99 !...but you cannot have a pure get with MPI
100  call mpp_error( fatal, 'MPP_TRANSMIT: you cannot transmit to ANY_PE using MPI.' )
101  else if( to_pe.NE.null_pe )then !no other valid cases except NULL_PE
102  call mpp_error( fatal, 'MPP_TRANSMIT: invalid to_pe.' )
103  end if
104 
105 !do the get
106  if( from_pe.GE.0 .AND. from_pe.LT.npes )then
107 !receive from from_pe
108  if( debug .and. (current_clock.NE.0) )call system_clock(start_tick)
109  if( block_comm ) then
110  call mpi_recv( get_data, get_len, mpi_type_, from_pe, comm_tag, mpp_comm_private, stat, error )
111  call mpi_get_count( stat, mpi_type_, rsize, error)
112  if(rsize .NE. get_len) then
113  print*, "rsize, get_len=", rsize, get_len, mpp_pe(), from_pe
114  call mpp_error(fatal, "MPP_TRANSMIT: get_len does not match size of data received")
115  endif
116  else
117 ! if( request_recv(from_pe).NE.MPI_REQUEST_NULL )then !only one message from from_pe->pe in queue
118  ! if( debug )write( stderr(),* )'PE waiting for receiving', pe, from_pe
119 ! call MPI_WAIT( request_recv(from_pe), stat, error )
120 ! end if
121  if(PRESENT(recv_request)) then
122  call mpi_irecv( get_data, get_len, mpi_type_, from_pe, comm_tag, mpp_comm_private, &
123  recv_request, error )
124  else
125  cur_recv_request = cur_recv_request + 1
126  if( cur_recv_request > max_request ) call mpp_error(fatal, &
127  "MPP_TRANSMIT: cur_recv_request is greater than max_request, increase mpp_nml request_multiply")
128  call mpi_irecv( get_data, get_len, mpi_type_, from_pe, comm_tag, mpp_comm_private, &
129  request_recv(cur_recv_request), error )
130  size_recv(cur_recv_request) = get_len
131  type_recv(cur_recv_request) = mpi_type_
132  endif
133  endif
134  if( debug .and. (current_clock.NE.0) )call increment_current_clock( event_recv, get_len*mpp_type_bytelen_ )
135  else if( from_pe.EQ.any_pe )then
136 !receive from MPI_ANY_SOURCE
137  if( debug .and. (current_clock.NE.0) )call system_clock(start_tick)
138  call mpi_recv( get_data, get_len, mpi_type_, mpi_any_source, comm_tag, mpp_comm_private, stat, error )
139  if( debug .and. (current_clock.NE.0) )call increment_current_clock( event_recv, get_len*mpp_type_bytelen_ )
140  else if( from_pe.EQ.all_pes )then
141  call mpp_error( fatal, 'MPP_TRANSMIT: from_pe=ALL_PES has ambiguous meaning, and hence is not implemented.' )
142  else if( from_pe.NE.null_pe )then !only remaining valid choice is NULL_PE
143  call mpp_error( fatal, 'MPP_TRANSMIT: invalid from_pe.' )
144  end if
145 
146  if( debug )then
147  call system_clock(tick)
148  write( stdout_unit,'(a,i18,a,i6,a,2i6,2i8)' )&
149  'T=',tick, ' PE=',pe, ' MPP_TRANSMIT end: to_pe, from_pe, put_len, get_len=', to_pe, from_pe, put_len, get_len
150  end if
151  return
152  end subroutine mpp_transmit_

160  subroutine mpp_broadcast_( data, length, from_pe, pelist )
161 !this call was originally bundled in with mpp_transmit, but that doesn't allow
162 !broadcast to a subset of PEs. This version will, and mpp_transmit will remain
163 !backward compatible.
164  mpp_type_, intent(inout) :: data(*)
165  integer, intent(in) :: length, from_pe
166  integer, intent(in), optional :: pelist(:)
167  integer :: n, i, from_rank, stdout_unit
168 
169  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_BROADCAST: You must first call mpp_init.' )
170  n = get_peset(pelist); if( peset(n)%count.EQ.1 )return
171 
172  if( debug )then
173  call system_clock(tick)
174  write( stdout_unit,'(a,i18,a,i6,a,2i6,2i8)' )&
175  'T=',tick, ' PE=',pe, ' MPP_BROADCAST begin: from_pe, length=', from_pe, length
176  end if
177 
178  if( .NOT.any(from_pe.EQ.peset(current_peset_num)%list) ) &
179  call mpp_error( fatal, 'MPP_BROADCAST: broadcasting from invalid PE.' )
180 
181  if( debug .and. (current_clock.NE.0) )call system_clock(start_tick)
182  ! find the rank of from_pe in the pelist.
183  do i = 1, mpp_npes()
184  if(peset(n)%list(i) == from_pe) then
185  from_rank = i - 1
186  exit
187  endif
188  enddo
189  if( mpp_npes().GT.1 )call mpi_bcast( data, length, mpi_type_, from_rank, peset(n)%id, error )
190  if( debug .and. (current_clock.NE.0) )call increment_current_clock( event_broadcast, length*mpp_type_bytelen_ )
191  return
192  end subroutine mpp_broadcast_

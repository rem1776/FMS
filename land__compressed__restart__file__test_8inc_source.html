20 subroutine land_compressed_restart_file(nz, nt, debug_flag)
21 
22  integer, intent(in) :: nz
23  integer, intent(in) :: nt
24  logical, intent(in), optional :: debug_flag
25 
26  type(FmsNetcdfFile_t) :: fileobj
27  integer :: my_rank
28  integer :: num_ranks
29  integer, dimension(:), allocatable :: pelist
30  integer, dimension(:), allocatable :: npes_corner
31  integer, dimension(:), allocatable :: npes_nelems
32  integer :: nx
33  integer :: total_nx
34  real(kind=r8_kind), dimension(:,:), allocatable :: temperature_data
35  real(kind=r8_kind), dimension(:), allocatable :: cdim_data
36  real(kind=r8_kind), dimension(:), allocatable :: lev_data
37  integer(kind=i8_kind) :: out_chksum
38  integer :: ndims
39  integer, dimension(:), allocatable :: dim_sizes
40  integer(kind=i8_kind) :: in_chksum
41  logical :: debug
42  integer :: err
43  integer :: i
44  integer :: j
45  integer :: k
46  integer :: s
47 
48  !Tell which test is running.
49  call mpi_barrier(mpi_comm_world, err)
50  call mpi_check(err)
51  if (mpp_pe() .eq. 0) then
52  write(output_unit,'(/a)',advance="no") &
53  "Running land (compressed) restart file test ... "
54  endif
55  call mpi_barrier(mpi_comm_world, err)
56  call mpi_check(err)
57 
58  !Initialize.
59  if (present(debug_flag)) then
60  debug = debug_flag
61  else
62  debug = .false.
63  endif
64  call random_seed()
65 
66  !Initalize some data. Each rank will have (my_rank+1)*5
67  !data elements.
68  my_rank = mpp_pe()
69  num_ranks = mpp_npes()
70  allocate(pelist(num_ranks))
71  allocate(npes_corner(num_ranks))
72  allocate(npes_nelems(num_ranks))
73  nx = 1
74  do i = 1, npes
75  pelist(i) = i - 1
76  npes_corner(i) = nx
77  npes_nelems(i) = i*5
78  nx = nx + npes_nelems(i)
79  enddo
80  nx = npes_nelems(my_rank+1)
81  total_nx = sum(npes_nelems)
82  allocate(temperature_data(nx, nz))
83  do i = 1, nz
84  do j = 1, nx
85  call random_number(temperature_data(j, i))
86  temperature_data(j, i) = temperature_data(j, i) + 300.05
87  enddo
88  enddo
89 
90  !Write a restart file.
91  !Open a restart file and initialize the file object.
92  call open_check(open_file(fileobj, "land_compressed.nc", "overwrite", &
93  pelist=pelist, nc_format="netcdf4", is_restart=.true.))
94 
95  !Add some global attributes to the restart file.
96  call register_global_attribute(fileobj, "globalatt1", real(7., kind=r8_kind))
97  call register_global_attribute(fileobj, "globalatt2", real(4., kind=r4_kind))
98  call register_global_attribute(fileobj, "globalatt3", int(3, kind=i4_kind))
99  call register_global_attribute(fileobj, "globalatt4", int(2, kind=i8_kind))
100  call register_global_attribute(fileobj, "globalatt5", "some text", str_len=len_trim("some text"))
101 
102  !Add dimensions and corresponding variables to the file.
103  !cdim (compressed dimension).
104  call register_axis(fileobj, "cdim", npes_corner=npes_corner, npes_nelems=npes_nelems)
105  call register_field(fileobj, "cdim", "double", (/"cdim"/))
106  allocate(cdim_data(nx))
107  do i = 1, nx
108  cdim_data(i) = 4.178 + 3.9567*i
109  enddo
110  call write_data(fileobj, "cdim", cdim_data)
111  call register_variable_attribute(fileobj, "cdim", "units", "compressed value", str_len=len_trim("compressed value"))
112 
113  !Height.
114  call register_axis(fileobj, "lev", nz)
115  call register_field(fileobj, "lev", "double", (/"lev"/))
116  allocate(lev_data(nz))
117  do i = 1, nz
118  lev_data(i) = 0.1111 + 20.236*i
119  enddo
120  call write_data(fileobj, "lev", lev_data)
121  call register_variable_attribute(fileobj, "lev", "units", "mb", str_len=len_trim("mb"))
122 
123  !Add a restart variable.
124  call register_restart_field(fileobj, "temperature", temperature_data, (/"cdim","lev "/))
125  call register_variable_attribute(fileobj, "temperature", "units", "K", str_len=len_trim("K"))
126 
127  !Perform a "simulation".
128  do i = 1, nt
129  do j = 1, nz
130  do k = 1, nx
131  temperature_data(k, j) = temperature_data(k, j) + 0.5*mpp_pe()
132  enddo
133  enddo
134  enddo
135 
136  !Calculate a checksum, that will be used to check that the data is
137  !read back in correctly.
138  out_chksum = mpp_chksum(temperature_data, pelist=(/mpp_pe()/))
139 
140  !Write out the restart file.
141  call write_restart(fileobj)
142 
143  !Close the file and destroy the file object.
144  call close_file(fileobj)
145  call mpi_barrier(mpi_comm_world, err)
146  call mpi_check(err)
147 
148  !Prepare to read the data back in.
149  deallocate(cdim_data)
150  deallocate(lev_data)
151  deallocate(temperature_data)
152 
153  !Read restart with an explicit axis register call.
154  !Re-open the restart file and re-initialize the file object.
155  call open_check(open_file(fileobj, "land_compressed.nc", "read", pelist=pelist, &
156  nc_format="netcdf4", is_restart=.true.))
157 
158  !Read the cdim variable back in and make sure it is the
159  !correct size.
160  ndims = get_variable_num_dimensions(fileobj, "cdim", broadcast=.true.)
161  allocate(dim_sizes(ndims))
162  call get_variable_size(fileobj, "cdim", dim_sizes, .true.)
163  if (dim_sizes(1) .ne. total_nx) then
164  call mpp_error(fatal, "compressed axis is not the correct size.")
165  endif
166  deallocate(dim_sizes)
167 
168  !Allocate a buffer and associate it with the
169  !temperature variable.
170  ndims = get_variable_num_dimensions(fileobj, "temperature", broadcast=.true.)
171  allocate(dim_sizes(ndims))
172  call get_variable_size(fileobj, "temperature", dim_sizes, broadcast=.true.)
173  allocate(temperature_data(dim_sizes(1), dim_sizes(2)))
174  deallocate(dim_sizes)
175  temperature_data(:, :) = 0.
176 
177  !Read in the restart variable and close the file.
178  call read_data(fileobj, "temperature", temperature_data)
179  call close_file(fileobj)
180  call mpi_barrier(mpi_comm_world, err)
181  call mpi_check(err)
182 
183  !Check to see if the read-in data matches the data that was
184  !written out.
185  s = npes_corner(my_rank+1)
186  in_chksum = mpp_chksum(temperature_data(s:s+nx-1,:), pelist=(/mpp_pe()/))
187  call chksum_match(out_chksum, in_chksum, "temperature", debug)
188  call mpi_barrier(mpi_comm_world, err)
189  call mpi_check(err)
190  if (mpp_pe() .eq. 0) then
191  write(output_unit,'(a/)') trim(green)//"passed."//trim(color_end)
192  endif
193  call mpi_barrier(mpi_comm_world, err)
194  call mpi_check(err)
195 end subroutine land_compressed_restart_file

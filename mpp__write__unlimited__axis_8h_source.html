19  subroutine mpp_write_unlimited_axis_1d_(unit,field,domain,data,nelems_io)
20  integer, intent(in) :: unit
21  type(fieldtype), intent(inout) :: field
22  type(domain2D), intent(inout) :: domain
23  mpp_type_, intent(inout) :: data(:)
24  integer, intent(in) :: nelems_io(:) ! number of compressed elements from each
25  ! member of the io_domain. It MUST have the
26  ! same order as the io_domain pelist.
27  integer, allocatable :: pelist(:)
28  integer :: i,j,nelems,npes
29  type(domain2d), pointer :: io_domain=>null()
30  mpp_type_, allocatable, dimension(:) :: rbuff
31 
32  call mpp_clock_begin(mpp_write_clock)
33 
34  if( .NOT.module_is_initialized )call mpp_error( fatal, 'MPP_WRITE_UNLIMITED_AXIS_1D_: must first call mpp_io_init.' )
35  if( .NOT.mpp_file(unit)%valid )call mpp_error( fatal, 'MPP_WRITE_UNLIMITED_AXIS_1D_: invalid unit number.' )
36 
37  io_domain=>mpp_get_io_domain(domain)
38  if (.not. ASSOCIATED(io_domain)) call mpp_error( fatal, 'MPP_WRITE_UNLIMITED_AXIS_1D_: io_domain must be defined.' )
39  npes = mpp_get_domain_npes(io_domain)
40  allocate(pelist(npes))
41  call mpp_get_pelist(io_domain,pelist)
42 
43  nelems = sum(nelems_io(:))
44 
45  if(mpp_file(unit)%write_on_this_pe) allocate(rbuff(nelems))
46 
47  ! Note that the gatherV implied here is asymmetric; only root needs to know the vector of recv size
48  call mpp_gather(data,size(data),rbuff,nelems_io(:),pelist)
49 
50  if(mpp_file(unit)%write_on_this_pe) then
51  field%size(1) = nelems ! Correct the field size now that we have all the data
52  call write_record(unit, field, nelems, rbuff)
53  deallocate(rbuff)
54  endif
55  deallocate(pelist)
56 
57  call mpp_clock_end(mpp_write_clock)
58  return
59  end subroutine mpp_write_unlimited_axis_1d_

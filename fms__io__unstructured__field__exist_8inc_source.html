22 !>Return a flag indicating whether the inputted field exists in the inputted
23 !!file, where the file is associated with an unstructured mpp domain.
24 function fms_io_unstructured_field_exist(file_name, &
25  field_name, &
26  domain) &
27  result(does_field_exist)
28 
29  !Inputs/Outputs
30  character(len=*),intent(in) :: file_name !<Name of a file.
31  character(len=*),intent(in) :: field_name !<Name of a field.
32  type(domainUG),intent(in) :: domain !<An unstructured mpp domain.
33  logical(INT_KIND) :: does_field_exist !<Flag telling if the inputted field exists in the inputted file.
34 
35  !Local variables
36  logical(INT_KIND) :: file_exist !<Flag telling if the inputted file or one of its variants exists.
37  character(len=256) :: fname !<Actual name of the found file.
38  logical(INT_KIND) :: read_dist !<Flag telling if the file is "distributed" (has IO domain tile id appended onto the end).
39  integer(INT_KIND) :: funit !<A file unit.
40  integer(INT_KIND) :: nfile !<Index of the inputted file in the "files_read" module array.
41  integer(INT_KIND) :: i !<Loop variable.
42  integer(INT_KIND) :: ndim !<Number of dimensions in a file.
43  integer(INT_KIND) :: nvar !<Number of fields in a file.
44  integer(INT_KIND) :: natt !<Number of attributes in a file.
45  integer(INT_KIND) :: ntime !<Number of time levels in a file.
46  character(len=64) :: tmp_name !<Name of a field.
47  type(fieldtype),dimension(:),allocatable :: fields !<An array of fields found in the input file.
48 
49 
50  !Set default return value for the function.
51  does_field_exist = .false.
52 
53  !Return if the inputted field name is in valid.
54  if (len_trim(field_name) .eq. 0) then
55  return
56  endif
57  if (field_name(1:1) .eq. ' ') then
58  return
59  endif
60 
61  !Check if the file exists.
62  file_exist = fms_io_unstructured_get_file_name(file_name, &
63  fname, &
64  read_dist, &
65  domain)
66 
67  if (file_exist) then
68 
69  !Get the file unit for the input file.
70  call fms_io_unstructured_get_file_unit(fname, &
71  funit, &
72  nfile, &
73  read_dist, &
74  domain)
75 
76  !Get the number of dimensions, fields, attributes and time levels
77  !for the file.
78  call mpp_get_info(funit, &
79  ndim, &
80  nvar, &
81  natt, &
82  ntime)
83 
84  !Create an array of all fields contained in the file.
85  allocate(fields(nvar))
86  call mpp_get_fields(funit, &
87  fields)
88 
89  !Loop through the fields to see if the inputted field name matches
90  !any of the fields from the file.
91  do i = 1,nvar
92  call mpp_get_atts(fields(i), &
93  name=tmp_name)
94  if (lowercase(trim(tmp_name)) .eq. lowercase(trim(field_name))) then
95  does_field_exist = .true.
96  endif
97  enddo
98 
99  !Deallocate local allocatable.
100  deallocate(fields)
101  endif
102 
103  return
104 end function fms_io_unstructured_field_exist

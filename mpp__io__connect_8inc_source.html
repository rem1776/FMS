22 ! <SUBROUTINE NAME="mpp_open">
23 
24 ! <OVERVIEW>
25 ! Open a file for parallel I/O.
26 ! </OVERVIEW>
27 ! <DESCRIPTION>
28 ! Open a file for parallel I/O.
29 ! </DESCRIPTION>
30 ! <TEMPLATE>
31 ! call mpp_open( unit, file, action, form, access, threading, fileset,
32 ! iospec, nohdrs, recl, pelist )
33 ! </TEMPLATE>
34 
35 ! <OUT NAME="unit" TYPE="integer">
36 ! unit is intent(OUT): always _returned_by_ mpp_open().
37 ! </OUT>
38 ! <IN NAME="file" TYPE="character(len=*)">
39 ! file is the filename: REQUIRED
40 ! we append .nc to filename if it is a netCDF file
41 ! we append .<pppp> to filename if fileset is private (pppp is PE number)
42 ! </IN>
43 ! <IN NAME="action" TYPE="integer">
44 ! action is one of MPP_RDONLY, MPP_APPEND, MPP_WRONLY or MPP_OVERWR.
45 ! </IN>
46 ! <IN NAME="form" TYPE="integer">
47 ! form is one of MPP_ASCII: formatted read/write
48 ! MPP_NATIVE: unformatted read/write with no conversion
49 ! MPP_IEEE32: unformatted read/write with conversion to IEEE32
50 ! MPP_NETCDF: unformatted read/write with conversion to netCDF
51 ! </IN>
52 ! <IN NAME="access" TYPE="integer">
53 ! access is one of MPP_SEQUENTIAL or MPP_DIRECT (ignored for netCDF).
54 ! RECL argument is REQUIRED for direct access IO.
55 ! </IN>
56 ! <IN NAME="threading" TYPE="integer">
57 ! threading is one of MPP_SINGLE or MPP_MULTI
58 ! single-threaded IO in a multi-PE run is done by PE0.
59 ! </IN>
60 ! <IN NAME="fileset" TYPE="integer">
61 ! fileset is one of MPP_MULTI and MPP_SINGLE
62 ! fileset is only used for multi-threaded I/O
63 ! if all I/O PEs in <pelist> use a single fileset, they write to the same file
64 ! if all I/O PEs in <pelist> use a multi fileset, they each write an independent file
65 ! </IN>
66 ! <IN NAME="pelist" TYPE="integer">
67 ! pelist is the list of I/O PEs (currently ALL).
68 ! </IN>
69 ! <IN NAME="recl" TYPE="integer">
70 ! recl is the record length in bytes.
71 ! </IN>
72 ! <IN NAME="iospec" TYPE="character(len=*)">
73 ! iospec is deprecated, but included for interface uniformity reasons.
74 ! </IN>
75 ! <IN NAME="nohdrs" TYPE="logical">
76 ! nohdrs has no effect when action=MPP_RDONLY|MPP_APPEND or when form=MPP_NETCDF
77 ! </IN>
78 ! <NOTE>
79 ! The integer parameters to be passed as flags (<TT>MPP_RDONLY</TT>,
80 ! etc) are all made available by use association. The <TT>unit</TT>
81 ! returned by <TT>mpp_open</TT> is guaranteed unique. For non-netCDF I/O
82 ! it is a valid fortran unit number and fortran I/O can be directly called
83 ! on the file.
84 !
85 ! <TT>MPP_WRONLY</TT> will guarantee that existing files named
86 ! <TT>file</TT> will not be clobbered. <TT>MPP_OVERWR</TT>
87 ! allows overwriting of files.
88 !
89 ! Files opened read-only by many processors will give each processor
90 ! an independent pointer into the file, i.e:
91 !
92 ! <PRE>
93 ! namelist / nml / ...
94 ! ...
95 ! call mpp_open( unit, 'input.nml', action=MPP_RDONLY )
96 ! read(unit,nml)
97 ! </PRE>
98 !
99 ! will result in each PE independently reading the same namelist.
100 !
101 ! Metadata identifying the file and the version of
102 ! <TT>mpp_io_mod</TT> are written to a file that is opened
103 ! <TT>MPP_WRONLY</TT> or <TT>MPP_OVERWR</TT>. If this is a
104 ! multi-file set, and an additional global attribute
105 ! <TT>NumFilesInSet</TT> is written to be used by post-processing
106 ! software.
107 !
108 ! If <TT>nohdrs=.TRUE.</TT> all calls to write attributes will
109 ! return successfully <I>without</I> performing any writes to the
110 ! file. The default is <TT>.FALSE.</TT>.
111 !
112 ! For netCDF files, headers are always written even if
113 ! <TT>nohdrs=.TRUE.</TT>
114 ! </SUBROUTINE>
115 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
116 ! !
117 ! OPENING AND CLOSING FILES: mpp_open() and mpp_close() !
118 ! !
119 ! mpp_open( unit, file, action, form, access, threading, & !
120 ! fileset, iospec, nohdrs, recl, pelist ) !
121 ! integer, intent(out) :: unit !
122 ! character(len=*), intent(in) :: file !
123 ! integer, intent(in), optional :: action, form, access, threading, !
124 ! fileset, recl !
125 ! character(len=*), intent(in), optional :: iospec !
126 ! logical, intent(in), optional :: nohdrs !
127 ! integer, optional, intent(in) :: pelist(:) !default ALL !
128 ! !
129 ! unit is intent(OUT): always _returned_by_ mpp_open() !
130 ! file is the filename: REQUIRED !
131 ! we append .nc to filename if it is a netCDF file !
132 ! we append .<pppp> to filename if fileset is private (pppp is PE number) !
133 ! iospec is a deprecated option !
134 ! if nohdrs is .TRUE. headers are not written on non-netCDF writes. !
135 ! nohdrs has no effect when action=MPP_RDONLY|MPP_APPEND !
136 ! or when form=MPP_NETCDF !
137 ! FLAGS: !
138 ! action is one of MPP_RDONLY, MPP_APPEND or MPP_WRONLY !
139 ! form is one of MPP_ASCII: formatted read/write !
140 ! MPP_NATIVE: unformatted read/write, no conversion !
141 ! MPP_IEEE32: unformatted read/write, conversion to IEEE32 !
142 ! MPP_NETCDF: unformatted read/write, conversion to netCDF !
143 ! access is one of MPP_SEQUENTIAL or MPP_DIRECT (ignored for netCDF) !
144 ! RECL argument is REQUIRED for direct access IO !
145 ! threading is one of MPP_SINGLE or MPP_MULTI !
146 ! single-threaded IO in a multi-PE run is done by PE0 !
147 ! fileset is one of MPP_MULTI and MPP_SINGLE !
148 ! fileset is only used for multi-threaded I/O !
149 ! if all I/O PEs in <pelist> use a single fileset, !
150 ! they write to the same file !
151 ! if all I/O PEs in <pelist> use a multi fileset, !
152 ! they each write an independent file !
153 ! recl is the record length in bytes !
154 ! pelist is the list of I/O PEs (currently ALL) !
155 ! !
156 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
157  subroutine mpp_open( unit, file, action, form, access, threading, &
158  fileset, iospec, nohdrs, recl, &
159  iostat, is_root_pe, domain, &
160 !----------
161 !ug support
162  domain_ug)
163 !----------
164  integer, intent(out) :: unit
165  character(len=*), intent(in) :: file
166  integer, intent(in), optional :: action, form, access
167  integer, intent(in), optional :: threading, fileset, recl
168  character(len=*), intent(in), optional :: iospec
169  logical, intent(in), optional :: nohdrs
170  integer, intent(out), optional :: iostat
171  logical, intent(in), optional :: is_root_pe
172  type(domain2d), intent(in), optional :: domain
173 !----------
174 !ug support
175  type(domainUG),target,intent(in),optional :: domain_ug
176 !----------
177  character(len=16) :: act, acc, for, pos
178  character(len=128) :: mesg
179  character(len=256) :: text2
180  integer :: action_flag, form_flag, access_flag, threading_flag, fileset_flag, length
181  integer :: nfiles, tile_id(1), io_layout(2)
182  logical :: exists, on_root_pe, dist_file
183  logical :: write_on_this_pe, read_on_this_pe, io_domain_exist
184  integer :: ios, nc_pos !position of .nc in file name
185  type(axistype) :: unlim !used by netCDF with mpp_append
186 
187 !----------
188 !ug support
189  type(domain2d),pointer :: io_domain
190  type(domainUG),pointer :: io_domain_ug
191  integer(i4_kind) :: io_layout_ug
192  integer(i4_kind) :: tile_id_ug
193 !----------
194  integer*8 :: lenp
195  integer :: comm
196  integer :: info, ierror
197  integer,dimension(:), allocatable :: glist(:)
198  integer ::lena, lenb
199  character(len=12) ::ncblk
200  character(len=128) ::nc_name
201  integer ::f_size, f_stat
202  integer ::fsize, inital = 0
203  character(len=128) :: f_test
204 
205 !----------
206 !ug support
207  !Only allow one type of mpp domain.
208  if (present(domain) .and. present(domain_ug)) then
209  call mpp_error(FATAL, &
210  "mpp_open: domain and domain_ug cannot both be" &
211  //" present in the same mpp_open call.")
212  endif
213 
214  !Null initialize the unstructured I/O domain pointer.
215  io_domain => null()
216  io_domain_ug => null()
217 !----------
218 
219  call mpp_clock_begin(mpp_open_clock)
220  if( .NOT.module_is_initialized )call mpp_error( FATAL, 'MPP_OPEN: must first call mpp_io_init.' )
221  on_root_pe = mpp_pe() == mpp_root_pe()
222  if(present(is_root_pe)) on_root_pe = is_root_pe
223 
224  dist_file = .false.
225 !set flags
226  action_flag = MPP_WRONLY !default
227  if( PRESENT(action) )action_flag = action
228  form_flag = MPP_ASCII
229  if( PRESENT(form) )form_flag = form
230 #ifndef use_netCDF
231  if( form_flag.EQ.MPP_NETCDF ) &
232  call mpp_error( FATAL, 'MPP_OPEN: To open a file with form=MPP_NETCDF, you must compile mpp_io with -Duse_netCDF.' )
233 #endif
234  access_flag = MPP_SEQUENTIAL
235  if( PRESENT(access) )access_flag = access
236  threading_flag = MPP_SINGLE
237  if( npes.GT.1 .AND. PRESENT(threading) )threading_flag = threading
238  fileset_flag = MPP_MULTI
239  if( PRESENT(fileset) )fileset_flag = fileset
240  if( threading_flag.EQ.MPP_SINGLE )fileset_flag = MPP_SINGLE
241 
242 !-- iospec presence warning
243  if( PRESENT(iospec) ) &
244  & call mpp_error( NOTE, 'MPP_OPEN: iospec option has been deprecated and will be removed at some future date')
245 
246 
247  io_domain_exist = .false.
248  if( PRESENT(domain) ) then
249  io_domain => mpp_get_io_domain(domain)
250  if(associated(io_domain)) io_domain_exist = .true.
251 
252 !----------
253 !ug support
254  elseif (present(domain_ug)) then
255  io_domain_ug => mpp_get_UG_io_domain(domain_ug)
256  io_domain_exist = .true.
257 !----------
258 
259  endif
260 
261  write_on_this_pe = .true.
262  read_on_this_pe = .true.
263  if( threading_flag.EQ.MPP_SINGLE .AND. .NOT.on_root_pe ) then
264  write_on_this_pe = .false.
265  read_on_this_pe = .false.
266  endif
267  if(form_flag == MPP_NETCDF .AND. action_flag .NE. MPP_RDONLY) then
268  if(fileset_flag .EQ.MPP_SINGLE .AND. threading_flag.EQ.MPP_MULTI) then
269  call mpp_error(FATAL, "mpp_io_connect.inc(mpp_open): multiple thread and single "// &
270  "file writing/appending is not supported for netCDF file")
271  endif
272  if( fileset_flag.EQ.MPP_SINGLE .AND. .NOT.on_root_pe ) then
273  write_on_this_pe = .false.
274  read_on_this_pe = .false.
275  endif
276  endif
277 
278  if( io_domain_exist) then
279 
280 !----------
281 !ug support
282  if (associated(io_domain)) then
283  ! in this case, only write out from the root_pe of io_domain.
284  write_on_this_pe = mpp_domain_is_tile_root_pe(io_domain)
285  elseif (associated(io_domain_ug)) then
286  write_on_this_pe = mpp_domain_UG_is_tile_root_pe(io_domain_ug)
287  endif
288 !----------
289 
290  endif
291  if( action_flag == MPP_RDONLY) write_on_this_pe = .false.
292  !get a unit number
293  if( .NOT. write_on_this_pe .AND. action_flag.NE.MPP_RDONLY .AND. .NOT. io_domain_exist)then
294  unit = NULLUNIT !PEs not participating in IO from this mpp_open() will return this value for unit
295  call mpp_clock_end(mpp_open_clock)
296  return
297  end if
298  if( form_flag.EQ.MPP_NETCDF )then
299  do unit = maxunits+1,2*maxunits
300  if( .NOT.mpp_file(unit)%valid )exit
301  end do
302  if( unit.GT.2*maxunits ) then
303  write(mesg,*) 'all the units between ',maxunits+1,' and ',2*maxunits,' are used'
304  call mpp_error( FATAL, 'MPP_OPEN: too many open netCDF files.'//trim(mesg) )
305  endif
306  else
307  do unit = unit_begin, unit_end
308  inquire( unit,OPENED=mpp_file(unit)%opened )
309  if( .NOT.mpp_file(unit)%opened )exit
310  end do
311  if( unit.GT.unit_end ) then
312  write(mesg,*) 'all the units between ',unit_begin,' and ',unit_end,' are used'
313  call mpp_error( FATAL, 'MPP_OPEN: no available units.'//trim(mesg) )
314  endif
315  end if
316  mpp_file(unit)%valid = .true.
317  mpp_file(unit)%write_on_this_pe = write_on_this_pe
318  mpp_file(unit)%read_on_this_pe = read_on_this_pe
319  mpp_file(unit)%io_domain_exist = io_domain_exist
320  if( PRESENT(domain) ) then
321  allocate(mpp_file(unit)%domain)
322  mpp_file(unit)%domain = domain
323 
324 !----------
325 !ug support
326  elseif (present(domain_ug)) then
327  mpp_file(unit)%domain_ug => domain_ug
328 !----------
329 
330  endif
331 
332 !get a filename
333  nc_pos = index(file,'.nc.')
334  dist_file = nc_pos>0 ! this is a distributed file ending with filename.nc.0???
335  text = file
336  length = len_trim(file)
337  if(form_flag.EQ.MPP_NETCDF.AND. file(length-2:length) /= '.nc' .AND. .NOT.dist_file) &
338  text = trim(file)//'.nc'
339 
340 !----------
341 !ug support
342 !HELP: Is there any way to retrieve the I/O layout for an unstructured grid?
343 ! I could not find a way, so I added it into mpp_domains.
344  if (present(domain)) then
345  io_layout = mpp_get_io_domain_layout(domain)
346  elseif (present(domain_ug)) then
347  io_layout_ug = mpp_get_io_domain_UG_layout(domain_ug)
348  endif
349 !----------
350 
351  if( io_domain_exist) then
352 
353 !----------
354 !ug support
355  if (present(domain) .and. io_layout(1)*io_layout(2) .gt. 1) then
356  fileset_flag = MPP_MULTI
357  threading_flag = MPP_MULTI
358  tile_id = mpp_get_tile_id(io_domain)
359  text2 = trim(text)
360  if (tile_id(1) .ge. 10000) then
361  call mpp_error(FATAL, &
362  "mpp_open: tile_id should be less than" &
363  //" 10000 when io_domain exist")
364  endif
365  write(text,'(a,i4.4)') trim(text)//'.',tile_id(1)
366  if (action_flag .eq. MPP_RDONLY) then
367  inquire(file=trim(text),EXIST=exists)
368  if (.not. exists) then
369  write(text2,'(a,i6.6)') trim(text2)//'.',tile_id(1)
370  inquire(file=trim(text2),EXIST=exists)
371  if (.not.exists) then
372  call mpp_error(FATAL, &
373  "mpp_open: neither "// &
374  trim(text)//" nor "// &
375  trim(text2)//" exist and io" &
376  //" domain exist.")
377  endif
378  text = trim(text2)
379  endif
380  endif
381  elseif (present(domain_ug) .and. io_layout_ug .gt. 1) then
382  fileset_flag = MPP_MULTI
383  threading_flag = MPP_MULTI
384  tile_id_ug = mpp_get_UG_domain_tile_id(io_domain_ug)
385  text2 = trim(text)
386  if (tile_id_ug .ge. 10000) then
387  call mpp_error(FATAL, &
388  "mpp_open: tile_id should be less than" &
389  //" 10000 when io_domain exist")
390  endif
391  write(text,'(a,i4.4)') trim(text)//'.',tile_id_ug
392  if (action_flag .eq. MPP_RDONLY) then
393  inquire(file=trim(text),EXIST=exists)
394  if (.not. exists) then
395  write(text2,'(a,i6.6)') trim(text2)//'.',tile_id_ug
396  inquire(file=trim(text2),EXIST=exists)
397  if (.not.exists) then
398  call mpp_error(FATAL, &
399  "mpp_open: neither "// &
400  trim(text)//" nor "// &
401  trim(text2)//" exist and io" &
402  //" domain exist.")
403  endif
404  text = trim(text2)
405  endif
406  endif
407  else
408  fileset_flag = MPP_SINGLE
409  threading_flag = MPP_SINGLE
410  endif
411 !----------
412 
413  else if( fileset_flag.EQ.MPP_MULTI ) then
414  if(mpp_npes() > 10000) then
415  write( text,'(a,i6.6)' )trim(text)//'.', pe-mpp_root_pe()
416  else
417  write( text,'(a,i4.4)' )trim(text)//'.', pe-mpp_root_pe()
418  endif
419  endif
420  mpp_file(unit)%name = text
421  if( verbose )print '(a,2i6,x,a,5i5)', 'MPP_OPEN: PE, unit, filename, action, format, access, threading, fileset=', &
422  pe, unit, trim(mpp_file(unit)%name), action_flag, form_flag, access_flag, threading_flag, fileset_flag
423 
424 !action: read, write, overwrite, append: act and pos are ignored by netCDF
425  if( action_flag.EQ.MPP_RDONLY )then
426  act = 'READ'
427  pos = 'REWIND'
428  else if( action_flag.EQ.MPP_WRONLY .OR. action_flag.EQ.MPP_OVERWR )then
429  act = 'WRITE'
430  pos = 'REWIND'
431  else if( action_flag.EQ.MPP_APPEND )then
432  act = 'WRITE'
433  pos = 'APPEND'
434  else
435  call mpp_error( FATAL, 'MPP_OPEN: action must be one of MPP_WRONLY, MPP_APPEND or MPP_RDONLY.' )
436  end if
437 
438  mpp_file(unit)%threading = threading_flag
439  mpp_file(unit)%fileset = fileset_flag
440 
441  if( .NOT. write_on_this_pe .AND. action_flag.NE.MPP_RDONLY ) then
442  call mpp_clock_end(mpp_open_clock)
443  return
444  endif
445 
446 !access: sequential or direct: ignored by netCDF
447  if( form_flag.NE.MPP_NETCDF )then
448  if( access_flag.EQ.MPP_SEQUENTIAL )then
449  acc = 'SEQUENTIAL'
450  else if( access_flag.EQ.MPP_DIRECT )then
451  acc = 'DIRECT'
452  if( form_flag.EQ.MPP_ASCII )call mpp_error( FATAL, 'MPP_OPEN: formatted direct access I/O is prohibited.' )
453  if( .NOT.PRESENT(recl) ) &
454  call mpp_error( FATAL, 'MPP_OPEN: recl (record length in bytes) must be specified with access=MPP_DIRECT.' )
455  mpp_file(unit)%record = 1
456  records_per_pe = 1 !each PE writes 1 record per mpp_write
457  else
458  call mpp_error( FATAL, 'MPP_OPEN: access must be one of MPP_SEQUENTIAL or MPP_DIRECT.' )
459  end if
460  end if
461 
462 !threading: SINGLE or MULTI
463  if( threading_flag.EQ.MPP_MULTI )then
464 !fileset: MULTI or SINGLE (only for multi-threaded I/O
465  if( fileset_flag.EQ.MPP_SINGLE )then
466  if( form_flag.EQ.MPP_NETCDF .AND. act.EQ.'WRITE' ) &
467  call mpp_error( FATAL, 'MPP_OPEN: netCDF currently does not support single-file multi-threaded output.' )
468 
469  else if( fileset_flag.NE.MPP_MULTI )then
470  call mpp_error( FATAL, 'MPP_OPEN: fileset must be one of MPP_MULTI or MPP_SINGLE.' )
471  end if
472  else if( threading_flag.NE.MPP_SINGLE )then
473  call mpp_error( FATAL, 'MPP_OPEN: threading must be one of MPP_SINGLE or MPP_MULTI.' )
474  end if
475 
476 !open the file as specified above for various formats
477  if( form_flag.EQ.MPP_NETCDF )then
478 #ifdef PAR_ACCESS
479  !adding some items for netcdf-4.... --fmi
480  if( .NOT.PRESENT(pelist)) then
481  allocate (glist(0:npes-1))
482  call mpp_get_current_pelist(glist, name, comm)
483  end if
484 
485  !create info parts...
486  call MPI_INFO_CREATE(info, ierror)
487 
488  !F2003 convention
489  !call get_enviornment_variable('NC_BLKSZ', ncblk)
490  call GETENV( 'NC_BLKSZ', ncblk)
491  ncblk = trim(ncblk)
492 
493 
494  if (ncblk /= "") then
495 
496  call MPI_INFO_SET(info, "cb_buffer_size", ncblk, ierror)
497  call MPI_INFO_SET(info, "ind_rd_buffer_size", ncblk, ierror)
498  call MPI_INFO_SET(info, "ind_wr_buffer_size", ncblk, ierror)
499 
500  else
501  call MPI_INFO_SET(info, "ind_rd_buffer_size", "16777216", ierror)
502  call MPI_INFO_SET(info, "ind_wr_buffer_size", "16777216", ierror)
503  endif
504 
505 #else
506 !added by fmi to read NC_BLKSZ and NC_BLKSZ_filename...
507 
508 
509  !get regular nc_blksz...
510  !build env var for check
511  !write (*,*) 'hello', trim(mpp_file(unit)%name)
512  nc_name = 'NC_BLKSZ_'//trim(mpp_file(unit)%name)
513  !write (*,*) 'nc_name: ', nc_name, ' bcblk: ', ncblk
514 
515 
516 
517  !make the call.....
518  !f2003 replaces GETENV with get_enviornment_variable so the guts are here if we need to switch
519  !call get_enviornment_variable(trim(nc_name),ncblk )
520  call GETENV( trim(nc_name),ncblk )
521 
522  !might not be there...use the general setting
523 
524  if (ncblk .EQ. '') then
525  !call get_enviornment_variable( 'NC_BLKSZ', ncblk)
526  call GETENV( 'NC_BLKSZ', ncblk)
527 
528 
529  endif
530 
531  !if no general setting then use default
532  if (ncblk .EQ. '') then
533  ncblk = '64k' !change for platform...perhaps we should set an ifdef for this....
534  endif
535 
536  !set or convert the chunksize
537 
538  call file_size(ncblk, mpp_file(unit)%name, fsize)
539  !write (*,*) 'this is fsize after: ', fsize

915  subroutine file_size(fsize, fname, size)
916 
917  character(len=12), intent(in) ::fsize
918  character(len=128) ::filesize
919  character(len=128), intent(in),optional :: fname
920  character(len=128) :: filename
921  integer*4 :: fstat(13)

926  integer :: tend
927  logical :: there

929  size = 0
930 
931  filesize = fsize
932 
933  length = len(trim(fsize))
934  tend = length - 1
935 
936 
937 
938  if (filesize .EQ. 'file') then
939  filename = trim(fname)
940  INQUIRE( FILE=filename, EXIST=THERE )
941  if (THERE) then
942  ierr = stat(filename, fstat)
943  if (ierr .EQ. 0) then
944  size = fstat(8)
945  else
946  size = 0
947  end if
948  end if
949  elseif((filesize(length:length)>='a'.AND.fsize(length:length)<='z').OR.(filesize(length:length)>='A' &

952  READ(number, FMT='(I9)') size
953  if (filesize(length:length) >= 'a' .AND. fsize(length:length) <= 'z') then
954  filesize(length:length) = ACHAR ( ICHAR (filesize(length:length)) - 32)
955  end if
956  if ( filesize(length:length) .EQ. 'K') then
957  size = size*1024

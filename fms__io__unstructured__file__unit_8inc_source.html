22 !>Find the file unit for an inputted file, searching for its variants. If the
23 !!file is not found, then throw a fatal error.
24 subroutine fms_io_unstructured_file_unit(filename, &
25  funit, &
26  domain)
27 
28  !Inputs/Outputs
29  character(len=*),intent(in) :: filename !<The name of a file.
30  integer(INT_KIND),intent(out) :: funit !<The file unit for the input file.
31  type(domainUG),intent(in) :: domain !<An unstructured mpp domain.
32 
33  !Local variables
34  logical(INT_KIND) :: found_file !<Flag telling if the file exists.
35  character(len=256) :: actual_file !<Name of the found file.
36  logical(INT_KIND) :: read_dist !<Flag telling if the file is "distributed" (has IO domain tile id appended to the end).
37  integer(INT_KIND) :: nfile !<Index of the inputted file in the "files_read" module array.
38 
39  !Get the actual name of the file, searching for all possible variants. If
40  !the file is not found, then throw a fatal error.
41  found_file = fms_io_unstructured_get_file_name(filename, &
42  actual_file, &
43  read_dist, &
44  domain)
45  if (.not. found_file) then
46  call mpp_error(FATAL, &
47  "fms_io_unstructured_file_unit: file " &
48  //trim(filename)//" and corresponding variants are" &
49  //" not found.")
50  endif
51 
52  !Get the file unit for the input file.
53  call fms_io_unstructured_get_file_unit(actual_file, &
54  funit, &
55  nfile, &
56  read_dist, &
57  domain)
58 
59  return
60 end subroutine fms_io_unstructured_file_unit

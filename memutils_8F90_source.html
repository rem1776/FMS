29 !! Module to expose the memory printing API
30 !!
31 !! Model components at times desire to print the memory statics to stderr,
32 !! or another file (e.g. the log file). This can be useful in debugging.
33 !! This module exposes the print_memuse_stat and memutils_init calls for
34 !! use in external user code.
35 module memutils_mod
36 !Author: Balaji (V.Balaji@noaa.gov)
37  use mpp_mod, only: mpp_pe, mpp_root_pe, mpp_npes, mpp_min, mpp_max, mpp_sum, stderr
38  use mpp_memutils_mod, only: mpp_print_memuse_stats
39 
40  implicit none
41 
42  logical :: memutils_initialized=.false. !< Indicate if module has been initialized.
43  logical, private :: print_memory_usage=.false. !< Default behavior of print_memuse_stats()
44 
45  public :: memutils_init
46  public :: print_memuse_stats
47 
48 contains
49 
50  !> \brief Initialize the memutils module
51  !!
52  !! memutils_init initializes the print_memory_usage and memutils_initialized
53  !! module variables when called. This configures if print_memuse_stats() should
54  !! print the memory statistics when called. memutils_init, at present, can
55  !! be called multiple times, and potentially change the behavior of print_memuse_stats().
56  subroutine memutils_init(print_flag)
57  logical, optional :: print_flag !< Indicate if memory statistics should be printed by default
58 
59  if ( PRESENT(print_flag) ) print_memory_usage = print_flag
60  memutils_initialized = .true.
61  return
62  end subroutine memutils_init
63 
64 !> \brief Print memory usage stats to stdout, or a particular file
65 !!
66 !! API to allow external user code to print memory statistics to stderr, or an
67 !! optional file (Fortran file unit). The module variable `print_memory_usage`
68 !! will control the default behavior (set when memutils_init is called). The
69 !! default behavior can be modified passing in the optional `always` (logical)
70 !! parameter. If `always.eqv..TRUE.`, print_memuse_stats() will print the memory
71 !! statistics.
72  subroutine print_memuse_stats( text, unit, always )
73  character(len=*), intent(in) :: text !< Text to be printed before the memory statistics
74  integer, intent(in), optional :: unit !< Fortran file unit to where memory statistics should be recorded
75  logical, intent(in), optional :: always !< If `.TRUE.`, force memory statistics to be printed
76 
77  if( PRESENT(always) )then
78  if( .NOT.always )return
79  else
80  if( .NOT.print_memory_usage )return
81  end if
82  call mpp_print_memuse_stats(text, unit)
83  return
84  end subroutine print_memuse_stats
85 end module memutils_mod

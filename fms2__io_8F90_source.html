234 !< Namelist variables
235 integer :: ncchksz = 64*1024 !< User defined chunksize (in bytes) argument in netcdf file
236  !! creation calls. Replaces setting the NC_CHKSZ environment variable.
237 character (len = 10) :: netcdf_default_format = "64bit" !< User defined netcdf file format, acceptable values
238  !! are: "64bit", "classic", "netcdf4". This can be overwritten if you specify
239  !! "nc_format" in the open_file call
240 integer :: header_buffer_val = 16384 !< Use defined netCDF header buffer size(in bytes) used in
241  !! NF__ENDDEF
242 namelist / fms2_io_nml / &
243  ncchksz, netcdf_default_format, header_buffer_val
244 
245 contains
246 
247 !> @brief Reads the fms2_io_nml
248 subroutine fms2_io_init ()
249  integer :: mystat
250 
251 !> Check if the module has already been initialized
252  if (fms2_io_is_initialized) return
253 !> Call initialization routines that this module depends on
254  call mpp_init()
255  call mpp_domains_init()
256 !> Read the namelist
257  READ (input_nml_file, nml=fms2_io_nml, iostat=mystat)
258 !>Send the namelist variables to their respective modules
259  if (ncchksz .le. 0) then
260  call mpp_error(fatal, "ncchksz in fms2_io_nml must be a positive number.")
261  endif
262  call netcdf_io_init (ncchksz,header_buffer_val,netcdf_default_format)
263  if (header_buffer_val .le. 0) then
264  call mpp_error(fatal, "header_buffer_val in fms2_io_nml must be a positive number.")
265  endif
266  call blackboxio_init (ncchksz)
267 !> Mark the fms2_io as initialized
268  fms2_io_is_initialized = .true.
269 end subroutine fms2_io_init

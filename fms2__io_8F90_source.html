231 character (len = 10) :: netcdf_default_format = "64bit" !< User defined netcdf file format, acceptable values
232  !! are: "64bit", "classic", "netcdf4". This can be overwritten if you specify
233  !! "nc_format" in the open_file call

241 !> @brief Reads the fms2_io_nml
242 subroutine fms2_io_init ()
243  integer :: mystat
244 
245 !> Check if the module has already been initialized
246  if (fms2_io_is_initialized) return
247 !> Call initialization routines that this module depends on
248  call mpp_init()
249  call mpp_domains_init()
250 !> Read the namelist
251  READ (input_nml_file, nml=fms2_io_nml, iostat=mystat)
252 !>Send the namelist variables to their respective modules
253  if (ncchksz .le. 0) then
254  call mpp_error(fatal, "ncchksz in fms2_io_nml must be a positive number.")
255  endif
256  call netcdf_io_init (ncchksz,header_buffer_val,netcdf_default_format)
257  if (header_buffer_val .le. 0) then
258  call mpp_error(fatal, "header_buffer_val in fms2_io_nml must be a positive number.")
259  endif
260  call blackboxio_init (ncchksz)
261 !> Mark the fms2_io as initialized
262  fms2_io_is_initialized = .true.
263 end subroutine fms2_io_init

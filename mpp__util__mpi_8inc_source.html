28 subroutine mpp_error_basic( errortype, errormsg )
29  !a very basic error handler
30  !uses ABORT and FLUSH calls, may need to use cpp to rename
31 #ifdef __INTEL_COMPILER
32  ! Intel module containing tracebackQQ
33  use ifcore
34 #endif
35  integer, intent(in) :: errortype
36  character(len=*), intent(in), optional :: errormsg
37  character(len=512) :: text
38  logical :: opened
39  integer :: istat, errunit
40 
41  if( .NOT.module_is_initialized )call ABORT()
42 
43  select case( errortype )
44  case(NOTE)
45  text = 'NOTE' !just FYI
46  case(WARNING)
47  text = 'WARNING' !probable error
48  case(FATAL)
49  text = 'FATAL' !fatal error
50  case default
51  text = 'WARNING: non-existent errortype (must be NOTE|WARNING|FATAL)'
52  end select
53 
54  if( npes.GT.1 )write( text,'(a,i6)' )trim(text)//' from PE', pe !this is the mpp part
55  if( PRESENT(errormsg) )text = trim(text)//': '//trim(errormsg)
56 !$OMP CRITICAL (MPP_ERROR_CRITICAL)
57  select case( errortype )
58  case(NOTE)
59  if(pe==root_pe)write( out_unit,'(a)' )trim(text)
60  case default
61  errunit = stderr()
62  write( errunit, '(/a/)' )trim(text)
63  if(pe==root_pe)write( out_unit,'(/a/)' )trim(text)
64  if( errortype.EQ.FATAL .OR. warnings_are_fatal )then
65  FLUSH(out_unit)
66 #ifdef __INTEL_COMPILER
67  ! Get traceback and return quietly for correct abort
68  call TRACEBACKQQ(user_exit_code=-1)
69 #elif __GFORTRAN__
70  call BACKTRACE
71 #endif
72  call MPI_ABORT( MPI_COMM_WORLD, 1, error )
73  end if
74  end select
75 
76  error_state = errortype
77 !$OMP END CRITICAL (MPP_ERROR_CRITICAL)
78 
79 
80 end subroutine mpp_error_basic

82 !#####################################################################
83 !--- makes a PE set out of a PE list. A PE list is an ordered list of PEs
84 !--- a PE set is a triad (start,log2stride,size) for SHMEM, an a communicator for MPI
85 !--- if stride is non-uniform or not a power of 2,
86 !--- will return error (not required for MPI but enforced for uniformity)
87 function get_peset(pelist)
88  integer :: get_peset
89  integer, intent(in), optional :: pelist(:)
90  integer :: group, errunit
91  integer :: i, n, stride, l
92  integer, allocatable :: sorted(:)
93  character(len=128) :: text
94 
95  if( .NOT.PRESENT(pelist) )then !set it to current_peset_num
96  get_peset = current_peset_num; return
97  end if
98 
99  !--- first make sure pelist is monotonically increasing.
100  if (size(pelist(:)) .GT. 1) then
101  do n = 2, size(pelist(:))
102  if(pelist(n) <= pelist(n-1)) call mpp_error(FATAL, "GET_PESET: pelist is not monotonically increasing")
103  enddo
104  endif
105 
106  allocate( sorted(size(pelist(:))) )
107  sorted = pelist
108  errunit = stderr()
109  if( debug )write( errunit,* )'pelist=', pelist
110 
111  !find if this array matches any existing peset
112  do i = 1,peset_num
113  if( debug )write( errunit,'(a,3i6)' )'pe, i, peset_num=', pe, i, peset_num
114  if( size(sorted(:)).EQ.size(peset(i)%list(:)) )then
115  if( ALL(sorted.EQ.peset(i)%list) )then
116  deallocate(sorted)
117  get_peset = i; return
118  end if
119  end if
120  end do
121  !not found, so create new peset
122  peset_num = peset_num + 1
123  if( peset_num > current_peset_max ) call expand_peset()
124  i = peset_num !shorthand
125  !create list
126  allocate( peset(i)%list(size(sorted(:))) )
127  peset(i)%list(:) = sorted(:)
128  peset(i)%count = size(sorted(:))
129 
130  call MPI_GROUP_INCL( peset(current_peset_num)%group, size(sorted(:)), sorted-mpp_root_pe(), peset(i)%group, error )
131  call MPI_COMM_CREATE_GROUP(peset(current_peset_num)%id, peset(i)%group, &
132  DEFAULT_TAG, peset(i)%id, error )
133  deallocate(sorted)
134  get_peset = i
135 
136  return
137 
138 end function get_peset

142 subroutine mpp_sync( pelist, do_self )
143  integer, intent(in), optional :: pelist(:)
144  logical, intent(in), optional :: do_self
145  logical :: dself
146  integer :: n
147 
148  dself=.true.; if(PRESENT(do_self))dself=do_self
149 ! if(dself)call mpp_sync_self(pelist)
150 
151  n = get_peset(pelist); if( peset(n)%count.EQ.1 )return
152 
153  if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
154  call MPI_BARRIER( peset(n)%id, error )
155 
156  if( debug .and. (current_clock.NE.0) )call increment_current_clock(EVENT_WAIT)
157 
158  return

161 !#######################################################################
162  !this is to check if current PE's outstanding puts are complete
163  !but we can't use shmem_fence because we are actually waiting for
164  !a remote PE to complete its get
165 subroutine mpp_sync_self( pelist, check, request, msg_size, msg_type)
166  integer, intent(in), optional :: pelist(:)
167  integer, intent(in), optional :: check
168  integer, intent(inout), optional :: request(:)
169  integer, intent(in ), optional :: msg_size(:)
170  integer, intent(in ), optional :: msg_type(:)
171 
172  integer :: i, m, n, stride, my_check, rsize
173 
174  if( debug .and. (current_clock.NE.0) )call SYSTEM_CLOCK(start_tick)
175  my_check = EVENT_SEND
176  if(present(check)) my_check = check
177  if( my_check .NE. EVENT_SEND .AND. my_check .NE. EVENT_RECV ) then
178  call mpp_error( FATAL, 'mpp_sync_self: The value of optional argument check should be EVENT_SEND or EVENT_RECV')
179  endif
180 
181  if(PRESENT(request)) then
182  if( .not. present(check) ) then
183  call mpp_error(FATAL, 'mpp_sync_self: check is not present when request is present')
184  endif
185  if( my_check == EVENT_RECV ) then
186  if( .not. present(msg_size) ) then
187  call mpp_error(FATAL, 'mpp_sync_self: msg_size is not present when request is present and it is EVENT_RECV')
188  endif
189  if( .not. present(msg_type) ) then
190  call mpp_error(FATAL, 'mpp_sync_self: msg_type is not present when request is present and it is EVENT_RECV')
191  endif
192  if(size(msg_size) .NE. size(request)) then
193  call mpp_error(FATAL, 'mpp_sync_self: dimension mismatch between msg_size and request')
194  endif
195  if(size(msg_type) .NE. size(request)) then
196  call mpp_error(FATAL, 'mpp_sync_self: dimension mismatch between msg_type and request')
197  endif
198 
199  do m = 1, size(request(:))
200  if( request(m) == MPI_REQUEST_NULL ) cycle
201  call MPI_WAIT(request(m), stat, error )
202  call MPI_GET_COUNT(stat, msg_type(m), rsize, error)
203  if(msg_size(m) .NE. rsize) then
204  call mpp_error(FATAL, "mpp_sync_self: msg_size does not match size of data received")
205  endif
206  enddo
207  else
208  do m = 1, size(request(:))
209  if(request(m) .NE.MPI_REQUEST_NULL )call MPI_WAIT(request(m), stat, error )
210  enddo
211  endif
212  else
213  select case(my_check)
214  case(EVENT_SEND)
215  do m = 1,cur_send_request
216  if( request_send(m).NE.MPI_REQUEST_NULL )call MPI_WAIT( request_send(m), stat, error )
217  end do
218  cur_send_request = 0
219  case(EVENT_RECV)
220  do m = 1,cur_recv_request
221  call MPI_WAIT( request_recv(m), stat, error )
222  call MPI_GET_COUNT(stat, type_recv(m), rsize, error)
223  if(size_recv(m) .NE. rsize) then
224  call mpp_error(FATAL, "mpp_sync_self: size_recv does not match of data received")
225  endif
226  size_recv(m) = 0
227  end do
228  cur_recv_request = 0
229  end select
230  endif
231  if( debug .and. (current_clock.NE.0) )call increment_current_clock(EVENT_WAIT)
232  return
233 end subroutine mpp_sync_self

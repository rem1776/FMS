26 use mpp_mod, only : input_nml_file

28  axistype, mpp_get_axis_data
29 use mpp_domains_mod, only : mpp_domains_init, domain2d, mpp_define_layout, mpp_define_domains,&
30  mpp_global_sum, mpp_global_max, mpp_global_min, bitwise_exact_sum, mpp_get_compute_domain, &
31  mpp_domains_set_stack_size
32 use time_interp_external_mod, only : time_interp_external, time_interp_external_init,&
33  time_interp_external_exit, time_interp_external, init_external_field, get_external_field_size
34 use time_manager_mod, only : get_date, set_date, time_manager_init, set_calendar_type, julian, time_type, increment_time,&
35  noleap

37 use axis_utils_mod, only: get_axis_bounds

42 integer :: id, i, io_status, unit, ierr
43 character(len=128) :: filename='INPUT/aerosol.climatology.nc', fieldname='so4_anthro'
44 type(time_type) :: time
45 real, allocatable, dimension(:,:,:) :: data_d, data_g
46 logical, allocatable, dimension(:,:,:) :: mask_d
47 type(domain2d) :: domain, domain_out
48 integer :: layout(2), fld_size(4)
49 integer :: isc, iec, jsc, jec, isd, ied, jsd, jed
50 integer :: yy, mm, dd, hh, ss
51 real :: sm,mx,mn
52 character(len=12) :: cal_type
53 integer :: ntime=12,year0=1990,month0=1,day0=1,days_inc=31
54 type(horiz_interp_type) :: Hinterp
55 type(axistype) :: Axis_centers(4), Axis_bounds(4)
56 real :: lon_out(180,89), lat_out(180,89)
57 real, allocatable, dimension(:,:) :: lon_local_out, lat_local_out
58 real, allocatable, dimension(:) :: lon_in, lat_in
59 integer :: isc_o, iec_o, jsc_o, jec_o, outunit
60 
61 namelist /test_time_interp_ext_nml/ filename, fieldname,ntime,year0,month0,&
62  day0,days_inc, cal_type
63 
64 call constants_init
65 call mpp_init
66 call mpp_io_init
67 call mpp_domains_init
68 call time_interp_external_init
69 call time_manager_init
70 call horiz_interp_init
71 
72 #ifdef INTERNAL_FILE_NML
73  read (input_nml_file, test_time_interp_ext_nml, iostat=io_status)
74  ierr = check_nml_error(io_status, 'test_time_interp_ext_nml')
75 #else
76  unit = open_namelist_file()
77  ierr=1; do while (ierr /= 0)
78  read (unit, nml=test_time_interp_ext_nml, iostat=io_status, end=10)
79  ierr = check_nml_error(io_status, 'test_time_interp_ext_nml')
80  enddo
81 10 call close_file (unit)
82 #endif
83 
84 outunit = stdlog()
85 write(outunit,test_time_interp_ext_nml)
86 
87 select case (trim(cal_type))
88 case ('julian')
89  call set_calendar_type(julian)
90 case ('no_leap')
91  call set_calendar_type(noleap)
92 case default
93  call mpp_error(fatal,'invalid calendar type')
94 end select
95 
96 outunit = stdout()
97 write(outunit,*) 'INTERPOLATING NON DECOMPOSED FIELDS'
98 write(outunit,*) '======================================'
99 
100 call time_interp_external_init
101 
102 id = init_external_field(filename,fieldname,verbose=.true.)
103 
104 fld_size = get_external_field_size(id)
105 
106 allocate(data_g(fld_size(1),fld_size(2),fld_size(3)))
107 data_g = 0
108 
109 time = set_date(year0,month0,day0,0,0,0)
110 
111 do i=1,ntime
112  call time_interp_external(id,time,data_g,verbose=.true.)
113  sm = sum(data_g)
114  mn = minval(data_g)
115  mx = maxval(data_g)
116  write(outunit,*) 'sum= ', sm
117  write(outunit,*) 'max= ', mx
118  write(outunit,*) 'min= ', mn
119  time = increment_time(time,0,days_inc)
120 enddo
121 
122 call mpp_define_layout((/1,fld_size(1),1,fld_size(2)/),mpp_npes(),layout)
123 call mpp_define_domains((/1,fld_size(1),1,fld_size(2)/),layout,domain)
124 call mpp_get_compute_domain(domain,isc,iec,jsc,jec)
125 call mpp_get_compute_domain(domain,isd,ied,jsd,jed)
126 
127 call mpp_domains_set_stack_size(fld_size(1)*fld_size(2)*min(fld_size(3),1)*2)
128 allocate(data_d(isd:ied,jsd:jed,fld_size(3)))
129 data_d = 0
130 
131 write(outunit,*) 'INTERPOLATING DOMAIN DECOMPOSED FIELDS'
132 write(outunit,*) '======================================'
133 
134 id = init_external_field(filename,fieldname,domain=domain, verbose=.true.)
135 
136 time = set_date(year0,month0,day0)
137 
138 do i=1,ntime
139  call time_interp_external(id,time,data_d,verbose=.true.)
140  sm = mpp_global_sum(domain,data_d,flags=bitwise_exact_sum)
141  mx = mpp_global_max(domain,data_d)
142  mn = mpp_global_min(domain,data_d)
143  write(outunit,*) 'global sum= ', sm
144  write(outunit,*) 'global max= ', mx
145  write(outunit,*) 'global min= ', mn
146  time = increment_time(time,0,days_inc)
147 enddo
148 
149 write(outunit,*) 'INTERPOLATING DOMAIN DECOMPOSED FIELDS USING HORIZ INTERP'
150 write(outunit,*) '======================================'
151 
152 
153 ! define a global 2 degree output grid
154 
155 do i=1,180
156  lon_out(i,:) = 2.0*i*atan(1.0)/45.0
157 enddo
158 
159 do i=1,89
160  lat_out(:,i) = (i-45)*2.0*atan(1.0)/45.0
161 enddo
162 
163 call mpp_define_layout((/1,180,1,89/),mpp_npes(),layout)
164 call mpp_define_domains((/1,180,1,89/),layout,domain_out)
165 call mpp_get_compute_domain(domain_out,isc_o,iec_o,jsc_o,jec_o)
166 
167 id = init_external_field(filename,fieldname,domain=domain_out,axis_centers=axis_centers,&
168  verbose=.true., override=.true.)
169 
170 allocate (lon_local_out(isc_o:iec_o,jsc_o:jec_o))
171 allocate (lat_local_out(isc_o:iec_o,jsc_o:jec_o))
172 
173 lon_local_out(isc_o:iec_o,jsc_o:jec_o) = lon_out(isc_o:iec_o,jsc_o:jec_o)
174 lat_local_out(isc_o:iec_o,jsc_o:jec_o) = lat_out(isc_o:iec_o,jsc_o:jec_o)
175 
176 call get_axis_bounds(axis_centers(1), axis_bounds(1), axis_centers)
177 call get_axis_bounds(axis_centers(2), axis_bounds(2), axis_centers)
178 
179 allocate(lon_in(fld_size(1)+1))
180 allocate(lat_in(fld_size(2)+1))
181 
182 call mpp_get_axis_data(axis_bounds(1), lon_in) ; lon_in = lon_in*atan(1.0)/45
183 call mpp_get_axis_data(axis_bounds(2), lat_in) ; lat_in = lat_in*atan(1.0)/45
184 
185 call horiz_interp_new(hinterp,lon_in,lat_in, lon_local_out, lat_local_out, &
186  interp_method='bilinear')
187 
188 time = set_date(year0,month0,day0)
189 
190 deallocate(data_d)
191 allocate(data_d(isc_o:iec_o,jsc_o:jec_o,fld_size(3)))
192 allocate(mask_d(isc_o:iec_o,jsc_o:jec_o,fld_size(3)))
193 do i=1,ntime
194  data_d = 0
195  call time_interp_external(id,time,data_d,verbose=.true.,horz_interp=hinterp, mask_out=mask_d)
196  sm = mpp_global_sum(domain_out,data_d,flags=bitwise_exact_sum)
197  mx = mpp_global_max(domain_out,data_d)
198  mn = mpp_global_min(domain_out,data_d)
199  write(outunit,*) 'global sum= ', sm
200  write(outunit,*) 'global max= ', mx
201  write(outunit,*) 'global min= ', mn
202 
203  where(mask_d)
204  data_d = 1.0
205  elsewhere
206  data_d = 0.0
207  endwhere
208  sm = mpp_global_sum(domain_out,data_d,flags=bitwise_exact_sum)
209  write(outunit,*) 'n valid points= ', sm
210 
211  time = increment_time(time,0,days_inc)
212 enddo
213 
214 call horiz_interp_del(hinterp)
215 
216 
217 call time_interp_external_exit
218 
219 
220 call mpp_io_exit
221 call mpp_exit
222 stop
223 
224 end program test_time_interp_external

25  !> Updates data domain of 3D field whose computational domains have been computed
26  subroutine mpp_do_update_ad_3d_( f_addrs, domain, update, d_type, ke, flags)
27  integer(i8_kind), intent(in) :: f_addrs(:,:)
28  type(domain2d), intent(in) :: domain
29  type(overlapspec), intent(in) :: update
30  mpp_type_, intent(in) :: d_type ! creates unique interface
31  integer, intent(in) :: ke
32  integer, optional, intent(in) :: flags
33 
34  mpp_type_ :: field(update%xbegin:update%xend, update%ybegin:update%yend,ke)
35  pointer(ptr_field, field)
36  integer :: update_flags
37  type(overlap_type), pointer :: overPtr => null()
38 
39 !equate to mpp_domains_stack
40  mpp_type_ :: buffer(size(mpp_domains_stack(:)))
41  pointer( ptr, buffer )
42  integer :: buffer_pos
43 
44 !receive domains saved here for unpacking
45 !for non-blocking version, could be recomputed
46  integer, allocatable :: msg1(:), msg2(:)
47  logical :: send(8), recv(8), update_edge_only
48  integer :: to_pe, from_pe, pos, msgsize, msgsize_send
49  integer :: n, l_size, l, m, i, j, k
50  integer :: is, ie, js, je, tMe, dir
51  integer :: buffer_recv_size, nlist, outunit
52 
53 
54  outunit = stdout()
55  ptr = loc(mpp_domains_stack)
56  l_size = size(f_addrs,1)
57 
58  update_flags = xupdate+yupdate !default
59  if( PRESENT(flags) )update_flags = flags
60 
61  update_edge_only = btest(update_flags, edgeonly)
62  recv(1) = btest(update_flags,east)
63  recv(3) = btest(update_flags,south)
64  recv(5) = btest(update_flags,west)
65  recv(7) = btest(update_flags,north)
66  if( update_edge_only ) then
67  if( .NOT. (recv(1) .OR. recv(3) .OR. recv(5) .OR. recv(7)) ) then
68  recv(1) = .true.
69  recv(3) = .true.
70  recv(5) = .true.
71  recv(7) = .true.
72  endif
73  else
74  recv(2) = recv(1) .AND. recv(3)
75  recv(4) = recv(3) .AND. recv(5)
76  recv(6) = recv(5) .AND. recv(7)
77  recv(8) = recv(7) .AND. recv(1)
78  endif
79  send = recv
80 
81  if(debug_message_passing) then
82  nlist = size(domain%list(:))
83  allocate(msg1(0:nlist-1), msg2(0:nlist-1) )
84  msg1 = 0
85  msg2 = 0
86  do m = 1, update%nrecv
87  overptr => update%recv(m)
88  msgsize = 0
89  do n = 1, overptr%count
90  dir = overptr%dir(n)
91  if(recv(dir)) then
92  is = overptr%is(n); ie = overptr%ie(n)
93  js = overptr%js(n); je = overptr%je(n)
94  msgsize = msgsize + (ie-is+1)*(je-js+1)
95  end if
96  end do
97  from_pe = update%recv(m)%pe
98  l = from_pe-mpp_root_pe()
99  call mpp_recv( msg1(l), glen=1, from_pe=from_pe, block=.false., tag=comm_tag_1 )
100  msg2(l) = msgsize
101  enddo
102 
103  do m = 1, update%nsend
104  overptr => update%send(m)
105  msgsize = 0
106  do n = 1, overptr%count
107  dir = overptr%dir(n)
108  if(send(dir)) then
109  is = overptr%is(n); ie = overptr%ie(n)
110  js = overptr%js(n); je = overptr%je(n)
111  msgsize = msgsize + (ie-is+1)*(je-js+1)
112  end if
113  end do
114  call mpp_send( msgsize, plen=1, to_pe=overptr%pe, tag=comm_tag_1 )
115  enddo
116  call mpp_sync_self(check=event_recv)
117 
118  do m = 0, nlist-1
119  if(msg1(m) .NE. msg2(m)) then
120  print*, "My pe = ", mpp_pe(), ",domain name =", trim(domain%name), ",from pe=", &
121  domain%list(m)%pe, ":send size = ", msg1(m), ", recv size = ", msg2(m)
122  call mpp_error(fatal, "mpp_do_update: mismatch on send and recv size")
123  endif
124  enddo
125  call mpp_sync_self()
126  write(outunit,*)"NOTE from mpp_do_update: message sizes are matched between send and recv for domain " &
127  //trim(domain%name)
128  deallocate(msg1, msg2)
129  endif
130 
131  !recv
132  buffer_pos = 0
133  do m = 1, update%nrecv
134  overptr => update%recv(m)
135  if( overptr%count == 0 )cycle
136  msgsize = 0
137  do n = 1, overptr%count
138  dir = overptr%dir(n)
139  if(recv(dir)) then
140  tme = overptr%tileMe(n)
141  is = overptr%is(n); ie = overptr%ie(n)
142  js = overptr%js(n); je = overptr%je(n)
143  msgsize = msgsize + (ie-is+1)*(je-js+1)
144  msgsize_send = (ie-is+1)*(je-js+1)*ke*l_size
145  pos = buffer_pos + msgsize_send
146  do l=1,l_size ! loop over number of fields
147  ptr_field = f_addrs(l, tme)
148  do k = ke,1,-1
149  do j = je, js, -1
150  do i = ie, is, -1
151  buffer(pos) = field(i,j,k)
152  field(i,j,k) = 0.
153  pos = pos - 1
154  end do
155  end do
156  end do
157  end do
158  end if
159  end do
160 
161  msgsize = msgsize*ke*l_size
162  if( msgsize.GT.0 )then
163  to_pe = overptr%pe
164  call mpp_send( buffer(buffer_pos+1), plen=msgsize, to_pe=to_pe, tag=comm_tag_2 )
165  buffer_pos = buffer_pos + msgsize
166  end if
167  end do ! end do m = 1, update%nrecv
168  buffer_recv_size = buffer_pos
169 
170  ! send
171  do m = 1, update%nsend
172  overptr => update%send(m)
173  if( overptr%count == 0 )cycle
174  pos = buffer_pos
175  msgsize = 0
176  do n = 1, overptr%count
177  dir = overptr%dir(n)
178  if( send(dir) ) msgsize = msgsize + overptr%msgsize(n)
179  enddo
180  if( msgsize.GT.0 )then
181  msgsize = msgsize*ke*l_size
182  msgsize_send = msgsize
183  from_pe = overptr%pe
184  call mpp_recv( buffer(buffer_pos+1), glen=msgsize, from_pe=from_pe, block=.false., tag=comm_tag_2 )
185  buffer_pos = buffer_pos + msgsize
186  end if
187  end do ! end do ist = 0,nlist-1
188 
189  call mpp_sync_self(check=event_recv)
190 
191  buffer_pos = buffer_recv_size
192 
193  ! send
194  do m = 1, update%nsend
195  overptr => update%send(m)
196  if( overptr%count == 0 )cycle
197  pos = buffer_pos
198  msgsize = 0
199  do n = 1, overptr%count
200  dir = overptr%dir(n)
201  if( send(dir) ) msgsize = msgsize + overptr%msgsize(n)
202  enddo
203  if( msgsize.GT.0 )then
204  buffer_pos = pos
205  end if
206 
207  do n = 1, overptr%count
208  dir = overptr%dir(n)
209  if( send(dir) ) then
210  tme = overptr%tileMe(n)
211  is = overptr%is(n); ie = overptr%ie(n)
212  js = overptr%js(n); je = overptr%je(n)
213  select case( overptr%rotation(n) )
214  case(zero)
215  do l=1,l_size ! loop over number of fields
216  ptr_field = f_addrs(l, tme)
217  do k = 1,ke
218  do j = js, je
219  do i = is, ie
220  pos = pos + 1
221  field(i,j,k)=field(i,j,k)+buffer(pos)
222  end do
223  end do
224  end do
225  end do
226  case( minus_ninety )
227  do l=1,l_size ! loop over number of fields
228  ptr_field = f_addrs(l, tme)
229  do k = 1,ke
230  do i = is, ie
231  do j = je, js, -1
232  pos = pos + 1
233  field(i,j,k)=field(i,j,k)+buffer(pos)
234  end do
235  end do
236  end do
237  end do
238  case( ninety )
239  do l=1,l_size ! loop over number of fields
240  ptr_field = f_addrs(l, tme)
241  do k = 1,ke
242  do i = ie, is, -1
243  do j = js, je
244  pos = pos + 1
245  field(i,j,k)=field(i,j,k)+buffer(pos)
246  end do
247  end do
248  end do
249  end do
250  case( one_hundred_eighty )
251  do l=1,l_size ! loop over number of fields
252  ptr_field = f_addrs(l, tme)
253  do k = 1,ke
254  do j = je, js, -1
255  do i = ie, is, -1
256  pos = pos + 1
257  field(i,j,k)=field(i,j,k)+buffer(pos)
258  end do
259  end do
260  end do
261  end do
262  end select
263  endif
264  end do ! do n = 1, overPtr%count
265 
266  msgsize = pos - buffer_pos
267  if( msgsize.GT.0 )then
268  buffer_pos = pos
269  end if
270  end do ! end do ist = 0,nlist-1
271 
272  call mpp_sync_self()
273 
274  return
275  end subroutine mpp_do_update_ad_3d_
276 !> @}

22 program test_mosaic

65  "contact=", n, ": tile1=", tile1(n), " tile2=", tile2(n), &
66  " is1=", istart1(n), " ie1=", iend1(n), &
67  " js1=", jstart1(n), " je1=", jend1(n), &
68  " is2=", istart2(n), " ie2=", iend2(n), &
69  " js2=", jstart2(n), " je2=", jend2(n)
70 end do
71

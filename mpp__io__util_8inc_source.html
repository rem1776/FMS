67 ! copy global file attributes for use by user

95  subroutine mpp_get_field_atts(field, name, units, longname, min, max, missing, ndim, siz, axes, atts, &
96  valid, scale, add, checksum)
97 
98  type(fieldtype), intent(in) :: field
99  character(len=*), intent(out), optional :: name, units
100  character(len=*), intent(out), optional :: longname
101  real, intent(out), optional :: min,max,missing
102  integer, intent(out), optional :: ndim
103  integer, intent(out), dimension(:), optional :: siz
104  type(validtype), intent(out), optional :: valid
105  real, intent(out), optional :: scale
106  real, intent(out), optional :: add
107  integer(i8_kind), intent(out), dimension(:), optional :: checksum
108 
109  type(atttype), intent(inout), dimension(:), optional :: atts
110  type(axistype), intent(inout), dimension(:), optional :: axes
111 
112  integer :: n,m, check_exist
113 
114  if (PRESENT(name)) name = field%name
115  if (PRESENT(units)) units = field%units
116  if (PRESENT(longname)) longname = field%longname
117  if (PRESENT(min)) min = field%min
118  if (PRESENT(max)) max = field%max
119  if (PRESENT(missing)) missing = field%missing
120  if (PRESENT(ndim)) ndim = field%ndim
121  if (PRESENT(atts)) then
122  atts = default_att
123  n = size(atts(:));m=size(field%Att(:))
124  if (n.LT.m)&
125  call mpp_error(FATAL,'attribute array not large enough in mpp_get_field_atts, field '//&
126  trim(field%name))
127  do n=1,m
128  atts(n) = field%Att(n)
129  end do
130  end if
131  if (PRESENT(axes)) then
132  axes = default_axis
133  n = size(axes(:));m=field%ndim
134  if (n.LT.m) &
135  call mpp_error(FATAL,'axis array not large enough in mpp_get_field_atts, field '//&
136  trim(field%name))
137  do n=1,m
138  axes(n) = field%axes(n)
139  end do
140  end if
141  if (PRESENT(siz)) then
142  siz = -1
143  n = size(siz(:));m=field%ndim
144  if (n.LT.m) &
145  call mpp_error(FATAL,'size array not large enough in mpp_get_field_atts, field '//&
146  trim(field%name))
147  do n=1,m
148  siz(n) = field%size(n)
149  end do
150  end if
151 
152  if(PRESENT(valid)) then
153  call mpp_get_valid(field,valid)
154  endif
155 
156  if(PRESENT(scale)) scale = field%scale
157  if(present(add)) add = field%add
158  if(present(checksum)) then
159  checksum = 0
160  check_exist = mpp_find_att(field%Att(:),"checksum")
161  if ( check_exist >= 0 ) then
162  if(size(checksum(:)) >size(field%checksum(:))) call mpp_error(FATAL,"size(checksum(:)) >size(field%checksum(:))")
163  checksum = field%checksum(1:size(checksum(:)))
164  endif
165  endif
166 
167  return
168  end subroutine mpp_get_field_atts

596  !#####################################################################
597  ! based on presence/absence of attributes, defines valid range or missing
598  ! value. For details, see section 8.1 of NetCDF User Guide
599  subroutine mpp_get_valid(f,v)
600  type(fieldtype),intent(in) :: f ! field
601  type(validtype),intent(out) :: v ! validator
602 
603  integer :: irange,imin,imax,ifill,imissing,iscale
604  integer :: valid_T, scale_T ! types of attributes
605 
606  v%is_range = .true.
607  v%min = -HUGE(v%min); v%max = HUGE(v%max)
608  if (f%natt == 0) return
609  ! find indices of relevant attributes
610  irange = mpp_find_att(f%att,'valid_range')
611  imin = mpp_find_att(f%att,'valid_min')
612  imax = mpp_find_att(f%att,'valid_max')

614  imissing = mpp_find_att(f%att,'missing_value')
615 
616  ! find the widest type of scale and offset; note that the code
617  ! uses assumption that NetCDF types are arranged in th order of rank,
618  ! that is NF_BYTE < NF_SHORT < NF_INT < NF_FLOAT < NF_DOUBLE
619  scale_T = 0
620  iscale = mpp_find_att(f%att,'scale_factor')
621  if(iscale>0) scale_T = f%att(iscale)%type

624

627  valid_T = 0
628  if (irange>0) then
629  v%min = f%att(irange)%fatt(1)
630  v%max = f%att(irange)%fatt(2)
631  valid_T = f%att(irange)%type
632  else if (imax>0.or.imin>0) then
633  if(imax>0) then
634  v%max = f%att(imax)%fatt(1)
635  valid_T = max(valid_T,f%att(imax)%type)
636  endif
637  if(imin>0) then
638  v%min = f%att(imin)%fatt(1)
639  valid_T = max(valid_T,f%att(imin)%type)
640  endif
641  else if (imissing > 0) then
642  v%is_range = .false.
643  ! here we always scale, since missing_value is supposed to be in
644  ! external representation
645  v%min = f%att(imissing)%fatt(1)*f%scale + f%add
646  else if (ifill>0) then
647  !z1l ifdef is added in to be able to compile without using use_netCDF.
648 #ifdef use_netCDF
649  ! define min and max according to _FillValue
650  if(f%att(ifill)%fatt(1)>0) then
651  ! if _FillValue is positive, then it defines valid maximum
652  v%max = f%att(ifill)%fatt(1)
653  select case(f%type)
654  case (NF_BYTE,NF_SHORT,NF_INT)
655  v%max = v%max-1
656  case (NF_FLOAT)
657  v%max = nearest(nearest(real(v%max,4),-1.0),-1.0)
658  case (NF_DOUBLE)
659  v%max = nearest(nearest(real(v%max,8),-1.0),-1.0)
660  end select
661  ! always do the scaling, as the _FillValue is in external
662  ! representation
663  v%max = v%max*f%scale + f%add
664  else
665  ! if _FillValue is negative or zero, then it defines valid minimum
666  v%min = f%att(ifill)%fatt(1)
667  select case(f%type)
668  case (NF_BYTE,NF_SHORT,NF_INT)
669  v%min = v%min+1
670  case (NF_FLOAT)
671  v%min = nearest(nearest(real(v%min,4),+1.0),+1.0)
672  case (NF_DOUBLE)
673  v%min = nearest(nearest(real(v%min,8),+1.0),+1.0)
674  end select
675  ! always do the scaling, as the _FillValue is in external
676  ! representation
677  v%min = v%min*f%scale + f%add
678  endif
679 #endif
680  endif
681  ! If valid_range is the same type as scale_factor (actually the wider of
682  ! scale_factor and add_offset) and this is wider than the external data, then it
683  ! will be interpreted as being in the units of the internal (unpacked) data.
684  ! Otherwise it is in the units of the external (packed) data.
685  ! Note that it is not relevant if we went through the missing_data of _FillValue
686  ! brances, because in this case all irange, imin, and imax are less then 0
687  if(.not.((valid_T == scale_T).and.(scale_T>f%type))) then
688  if(irange>0 .or. imin>0) then
689  v%min = v%min*f%scale + f%add
690  endif
691  if(irange>0 .or. imax>0) then
692  v%max = v%max*f%scale + f%add
693  endif
694  endif
695 
696  end subroutine mpp_get_valid

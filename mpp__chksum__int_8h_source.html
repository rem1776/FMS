27 !> Calculates integer checksum over pelist
28 function mpp_chksum_int_( var, pelist, mask_val )
29  integer(i8_kind) :: mpp_chksum_int_
30  mpp_type_, intent(in) :: var mpp_rank_
31  integer, optional :: pelist(:)
32  mpp_type_, intent(in), optional :: mask_val
33 
34  if ( PRESENT(mask_val) ) then
35  !PACK on var/=mask_val ignores values in var
36  !equiv to setting those values=0, but on sparse arrays
37  !pack should return much smaller array to sum
38  mpp_chksum_int_ = sum( int( pack(var,var/=mask_val),i8_kind) )
39  else
40  mpp_chksum_int_ = sum(int(var,i8_kind))
41  end if
42 
43  call mpp_sum( mpp_chksum_int_, pelist )
44  return
45 
46  end function mpp_chksum_int_

49 !> Handles real mask for easier implementation
50 function mpp_chksum_int_rmask_( var, pelist, mask_val )
51  integer(KIND=i8_kind) :: mpp_chksum_int_rmask_
52  mpp_type_, intent(in) :: var mpp_rank_
53  integer, optional :: pelist(:)
54  real, intent(in) :: mask_val
55  integer(KIND(var))::imask_val
56  integer(KIND=i4_kind)::i4tmp(2)=0
57  real(kind=r4_kind)::r4tmp(2)=0
58  integer(KIND=i8_kind) :: i8tmp=0
59  !high fidelity error message
60  character(LEN=1) :: tmpstr1,tmpstr2,tmpstr3
61  character(LEN=32) :: tmpstr4
62  character(LEN=512) :: errstr
63 
64 ! Primary Logic: These first two are the "expected" branches.
65 !! These all resolve to MPP_FILL_INT
66  !!Should catch real "default_fill"(MPP_FILL_DOUBLE)
67  if (mask_val == mpp_fill_double ) then !this is FMS variable field default fill
68  ! we've packed an MPP_FILL_
69  imask_val = mpp_fill_int
70  !Current NETCDF fill values (AKA MPP_FILL_*) designed towards CEILING(MPP_FILL_{FLOAT,DOUBLE},kind=4byte)=MPP_FILL_INT
71  else if ( ceiling(mask_val, i4_kind) == mpp_fill_int ) then
72  ! we've also packed an MPP_FILL_
73  imask_val = mpp_fill_int
74 ! Secondary Logic:
75 !! We've done something dangerous
76  else
77  i8tmp = transfer(mask_val , i8tmp )
78  i4tmp = transfer(mask_val , i4tmp )
79  r4tmp = transfer(mask_val , r4tmp )
80  if ( i8tmp == mpp_fill_int ) then
81  ! we've packed an MPP_FILL_
82  imask_val = mpp_fill_int
83  else if ( any(i4tmp == mpp_fill_int) ) then
84  ! we've packed an MPP_FILL_
85  imask_val = mpp_fill_int
86  else if ( any(r4tmp == mpp_fill_double) ) then
87  ! we've packed an MPP_FILL_
88  imask_val = mpp_fill_int
89  else
90  ! we have no idea what this is
91  ! construct detailed errStr
92  errstr = "mpp_chksum: mpp_chksum_i"
93  write(unit=tmpstr1,fmt="(I1)") kind(var)
94  write(unit=tmpstr2,fmt="(I1)") SIZE(shape(var))
95  errstr = errstr // tmpstr1 // "_" // tmpstr2 // "d_rmask passed int var with REAL("
96  write(unit=tmpstr3,fmt="(I1)") kind(mask_val)
97  errstr = errstr // tmpstr3 // ") mask_val="
98  write(unit=tmpstr4,fmt=*) mask_val
99  errstr = errstr // trim(tmpstr4) // "has been called with these strange values. Check your KINDS, "// &
100  "_FillValue, pack and mask_val. "// &
101  "Hint: Try being explicit and using MPP_FILL_{INT,FLOAT,DOUBLE}. "// &
102  "Continuing by using the default MPP_FILL_INT. " // &
103  "THIS WILL BE FATAL IN THE FUTURE!"
104  call mpp_error(warning, trim(errstr) )
105 
106  imask_val = mpp_fill_int
107  end if
108  end if
109 
110  mpp_chksum_int_rmask_ = mpp_chksum(var,pelist,mask_val=imask_val)
111 
112  return
113 
114 end function mpp_chksum_int_rmask_
